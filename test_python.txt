#  XLWINGS FALLBACK WRAPPER (for .xls files or corrupted .xlsm)

class XlwingsCellWrapper:
    """Wrapper for xlwings cell to provide openpyxl-compatible interface"""

    def __init__(self, xw_cell):
        self._xw_cell = xw_cell

    @property
    def value(self):
        """Return calculated value, not formula"""
        try:
            formula = self._xw_cell.formula
            if formula:
                val = self._xw_cell.value
                if isinstance(val, str) and val.startswith('='):
                    return self._xw_cell.value
                return val
            return self._xw_cell.value
        except:
            return self._xw_cell.value

    @property
    def data_type(self):
        """Return cell type: 'f' for formula, 'n' for number, 's' for string, etc."""
        try:
            if self._xw_cell.formula:
                return 'f'
        except:
            pass
        val = self._xw_cell.value
        if val is None:
            return 'n'
        elif isinstance(val, (int, float)):
            return 'n'
        elif isinstance(val, str):
            if val.startswith('='):
                return 'f'
            return 's'
        return 'n'

    @property
    def has_style(self):
        """Check if cell has style (font, border, etc.)"""
        try:
            return bool(self._xw_cell.font.name or self._xw_cell.fill.patternType)
        except:
            return False

    @property
    def font(self):
        return self._xw_cell.font

    @property
    def border(self):
        return self._xw_cell.border

    @property
    def fill(self):
        return self._xw_cell.fill

    @property
    def number_format(self):
        return self._xw_cell.number_format

    @property
    def alignment(self):
        return self._xw_cell.alignment

    @property
    def coordinate(self):
        return self._xw_cell.address


class XlwingsSheetWrapper:
    """Wrapper for xlwings sheet to provide openpyxl-compatible cell access"""

    def __init__(self, xw_sheet):
        self._xw_sheet = xw_sheet
        self._cell_cache = {}

    def __getitem__(self, key):
        """Support ws['A1'] or ws[1, 1] style access"""
        if isinstance(key, str):
            if key not in self._cell_cache:
                self._cell_cache[key] = XlwingsCellWrapper(self._xw_sheet[key])
            return self._cell_cache[key]
        elif isinstance(key, tuple):
            row, col = key
            cell_key = f"{row}_{col}"
            if cell_key not in self._cell_cache:
                self._cell_cache[cell_key] = XlwingsCellWrapper(self._xw_sheet.cell(row=row, column=col))
            return self._cell_cache[cell_key]
        else:
            raise TypeError(f"Invalid key type: {type(key)}")

    def iter_rows(self):
        """Iterate over rows"""
        for row in self._xw_sheet.used_range.rows:
            yield [XlwingsCellWrapper(cell) for cell in row]

    @property
    def max_row(self):
        return self._xw_sheet.used_range.last_cell.row

    @property
    def max_column(self):
        return self._xw_sheet.used_range.last_cell.column

    @property
    def merged_cells(self):
        """Return merged cell ranges"""
        return self._xw_sheet.used_range.merge_cells

    @property
    def column_dimensions(self):
        return self._xw_sheet.used_range.column_dimensions


class XlwingsWorkbookWrapper:
    """Wrapper for xlwings workbook to provide openpyxl-compatible interface"""

    def __init__(self, xw_book, file_path: str):
        self._xw_book = xw_book
        self.path = file_path
        self._sheets_cache = {}

    @property
    def sheetnames(self) -> List[str]:
        return [s.name for s in self._xw_book.sheets]

    def __getitem__(self, sheet_name: str):
        if sheet_name not in self._sheets_cache:
            xw_sheet = self._xw_book.sheets[sheet_name]
            self._sheets_cache[sheet_name] = XlwingsSheetWrapper(xw_sheet)
        return self._sheets_cache[sheet_name]

    def remove(self, sheet):
        if isinstance(sheet, str):
            sheet_name = sheet
        else:
            sheet_name = sheet.title if hasattr(sheet, 'title') else sheet.name
        self._xw_book.sheets[sheet_name].delete()
        if sheet_name in self._sheets_cache:
            del self._sheets_cache[sheet_name]

    def create_sheet(self, title: str, index=None):
        new_sheet = self._xw_book.sheets.add(name=title, before=index)
        return XlwingsSheetWrapper(new_sheet)

    def close(self):
        try:
            self._xw_book.close()
        except Exception:
            pass

    def save(self, path=None):
        """Save the workbook while preserving external links"""
        try:
            # Disable screen updating and automatic calculation to speed up save
            self._xw_book.app.screen_updating = False
            self._xw_book.app.calculation = 'manual'

            # Set link update to manual (don't update links on open)
            try:
                self._xw_book.api.LinkFormat = False  # Disable automatic link update
            except:
                pass

            # Save the workbook
            if path:
                self._xw_book.save(path)
            else:
                self._xw_book.save()

            # Restore settings
            self._xw_book.app.screen_updating = True
            self._xw_book.app.calculation = 'automatic'

        except Exception as e:
            print(f"[DEBUG] xlwings save error: {e}")
            self._xw_book.app.screen_updating = True
            self._xw_book.app.calculation = 'automatic'
            raise


def ensure_file_closed(file_path: str) -> None:
    file_path = str(file_path)
    if not os.path.exists(file_path):
        return
    try:
        with open(file_path, "a"):
            pass
    except PermissionError as exc:
        raise PermissionError(
            f"File is open. Please close it to proceed: {file_path}"
        ) from exc


def load_excel_with_fallback(file_path: str, keep_vba: bool = False, data_only: bool = False, read_only: bool = False):
    """
    Load Excel file with fallback from openpyxl to xlwings.

    Returns: tuple (workbook, used_xlwings: bool)
    """
    import openpyxl

    # First try openpyxl
    try:
        wb = openpyxl.load_workbook(
            file_path,
            keep_vba=keep_vba,
            data_only=data_only,
            read_only=read_only
        )
        return wb, False
    except Exception as e:
        error_msg = str(e).lower()
        if "content_types" in error_msg or "no item named" in error_msg:
            print(f"[DEBUG] openpyxl failed, trying xlwings fallback for: {file_path}")
        else:
            raise

    # Fallback to xlwings
    try:
        xw_book = xw.Book(file_path)
        wrapper = XlwingsWorkbookWrapper(xw_book, file_path)
        return wrapper, True
    except Exception as e:
        raise RuntimeError(f"Failed to load with both openpyxl and xlwings: {e}")

def _apply_report_design(ws) -> None:
    """Apply colour-coded design to a copied Report sheet."""
    FILL_YELLOW = PatternFill("solid", fgColor="FFD966")
    FILL_GREEN = PatternFill("solid", fgColor="70AD47")
    FILL_BLUE = PatternFill("solid", fgColor="BDD7EE")
    FILL_LBLUE = PatternFill("solid", fgColor="DDEEF7")
    FILL_NONE = PatternFill(fill_type=None)

    BLUE_ROWS = {
        9, 35, 56, 77, 123, 144, 187, 253, 259, 306, 466, 587, 593, 614, 620,
        641, 647, 658, 664, 675, 686, 707, 713, 774, 830, 1429, 507, 512, 517,
        835, 1876, 907, 918, 924, 930, 937, 984, 1043, 951, 1015, 982, 1083,
        1075, 1079, 1081, 1116, 1253, 1259, 1270, 1281, 1297, 1323, 1339, 1360,
        1372, 1423, 1434, 2581, 1440, 1446, 1452, 1686, 1692, 1728, 1824, 1865,
        1882, 1893, 1921, 1927, 1933, 1939, 1945, 1946, 1951, 1956, 1961, 1966,
        1971, 1976, 1988, 2015, 2022, 2032, 2103, 2043, 2049, 2055, 2061, 2097,
        2112, 2123, 2144, 2260, 2271, 2323, 2339, 2400, 2406, 2412, 2418, 2424,
        2430, 2436, 2452, 2458, 2504, 2510, 2516, 2522, 2528, 2534, 2540, 2546,
        2552, 2558, 2564,
    }

    for row in ws.iter_rows():
        row_num = row[0].row
        label_a = str(row[0].value or '').strip()
        label_b = str(row[1].value or '').strip() if len(row) > 1 else ''
        label = label_a or label_b

        if label.startswith('****'):
            fill, bold, wht_text = FILL_YELLOW, True, False
        elif label.startswith('***'):
            fill, bold, wht_text = FILL_GREEN, True, True
        elif label.startswith('**'):
            fill, bold, wht_text = FILL_BLUE, True, False
        elif row_num in BLUE_ROWS:
            fill, bold, wht_text = FILL_LBLUE, True, False
        else:
            fill, bold, wht_text = FILL_NONE, False, False

        for cell in row:
            cell.fill = fill
            existing = cell.font
            cell.font = Font(
                name=existing.name or 'Calibri',
                size=existing.size or 10,
                bold=bold or existing.bold,
                italic=existing.italic,
                underline=existing.underline,
                color='FFFFFFFF' if wht_text else (existing.color or '00000000'),
            )
            val = cell.value
            if isinstance(val, (int, float)):
                col_letter = get_column_letter(cell.column)
                if col_letter in ('Q', 'R', 'S', 'T', 'U', 'V') and -5 < val < 5 and val != 0:
                    cell.number_format = '0.0%'
                else:
                    cell.number_format = '#,##0'

    ws.column_dimensions['B'].width = 45
    ws.freeze_panes = 'A8'
    print(f"[_apply_report_design]  * Design applied to sheet '{ws.title}'")

#  BREAK EXTERNAL LINKS
def break_external_links(file_path: str, wb_values=None) -> None:
    """
    Remove all external link references from a saved .xlsx/.xlsm file so that
    Excel no longer shows the "UNABLE TO REFRESH" / "Update links" prompt.
    Also injects cached <v> values for ALL formula cells that are missing them.
    """
    import zipfile
    import shutil
    import tempfile

    if _LXML_AVAILABLE:
        ET = _lxml_etree
        _USE_LXML = True
    else:
        import xml.etree.ElementTree as ET
        _USE_LXML = False

    NS = 'http://schemas.openxmlformats.org/spreadsheetml/2006/main'
    NS_REL = 'http://schemas.openxmlformats.org/package/2006/relationships'
    NS_CT = 'http://schemas.openxmlformats.org/package/2006/content-types'
    R_NS = 'http://schemas.openxmlformats.org/officeDocument/2006/relationships'

    EXT_LINK_TYPE = 'http://schemas.openxmlformats.org/officeDocument/2006/relationships/externalLink'
    EXT_LINK_CT = 'application/vnd.openxmlformats-officedocument.spreadsheetml.externalLink+xml'

    print(f"[break_external_links] Using {'lxml (robust)' if _USE_LXML else 'stdlib ET (fallback)'}")

    def _parse(data: bytes):
        if _USE_LXML:
            parser = ET.XMLParser(recover=True, remove_blank_text=False)
            return ET.fromstring(data, parser)
        else:
            return ET.fromstring(data)

    def _serialize(root) -> bytes:
        if _USE_LXML:
            return ET.tostring(root, xml_declaration=True, encoding='UTF-8', standalone=True)
        else:
            _ns_map = {
                '': NS,
                'r': R_NS,
                'mc': 'http://schemas.openxmlformats.org/markup-compatibility/2006',
                'x14ac': 'http://schemas.microsoft.com/office/spreadsheetml/2009/9/ac',
                'xr': 'http://schemas.microsoft.com/office/spreadsheetml/2014/revision',
                'xr2': 'http://schemas.microsoft.com/office/spreadsheetml/2015/revision2',
                'xr3': 'http://schemas.microsoft.com/office/spreadsheetml/2016/revision3',
            }
            for prefix, uri in _ns_map.items():
                try:
                    ET.register_namespace(prefix, uri)
                except Exception:
                    pass
            return ET.tostring(root, encoding='unicode').encode('utf-8')

    def _find_rel(root, tag):
        return root.findall(f'{{{NS_REL}}}{tag}')

    path = Path(file_path)
    if not path.exists():
        print(f"[break_external_links] File not found: {file_path}")
        return

    tmp_path = None
    try:
        tmp_fd, tmp_path = tempfile.mkstemp(suffix=path.suffix, dir=str(path.parent))
        os.close(tmp_fd)

        with zipfile.ZipFile(str(path), 'r') as zin, \
                zipfile.ZipFile(tmp_path, 'w', compression=zipfile.ZIP_DEFLATED) as zout:

            all_names = set(zin.namelist())
            ext_link_parts = set()
            ext_link_rels = set()

            # ── 1. Scan workbook.xml.rels to identify external link parts ──
            wb_rels_name = next(
                (n for n in all_names if n.lower().endswith('workbook.xml.rels')), None
            )

            if wb_rels_name:
                wb_rels_xml = zin.read(wb_rels_name)
                wb_rels_root = _parse(wb_rels_xml)
                for rel in _find_rel(wb_rels_root, 'Relationship'):
                    if rel.get('Type') == EXT_LINK_TYPE:
                        target = rel.get('Target', '')
                        full = 'xl/' + target.lstrip('/')
                        ext_link_parts.add(full)
                        head, tail = full.rsplit('/', 1)
                        ext_link_rels.add(f'{head}/_rels/{tail}.rels')

            # ── 2. BUILD CORRECT SHEET MAPPING ──────────────────────────
            # Maps "xl/worksheets/sheet3.xml" → "MZLZ Cost Control" (actual sheet name)
            xml_to_sheetname = {}
            wb_xml_name = next(
                (n for n in all_names if n.lower().endswith('workbook.xml')
                 and '/_rels/' not in n.lower()), None
            )

            if wb_xml_name and wb_rels_name:
                # Step A: workbook.xml.rels → rId → zip path
                rid_to_target = {}
                wb_rels_root2 = _parse(zin.read(wb_rels_name))
                for rel in _find_rel(wb_rels_root2, 'Relationship'):
                    rid = rel.get('Id', '')
                    target = rel.get('Target', '')
                    if 'worksheets/' in target:
                        target_clean = target.lstrip('/')
                        if target_clean.startswith('xl/'):
                            full_target = target_clean
                        else:
                            full_target = 'xl/' + target_clean
                        rid_to_target[rid] = full_target

                # Step B: workbook.xml → <sheet name="..." r:id="rId3"/>
                wb_root_for_map = _parse(zin.read(wb_xml_name))
                for sheet_el in wb_root_for_map.findall(f'{{{NS}}}sheets/{{{NS}}}sheet'):
                    sname = sheet_el.get('name')
                    rid = sheet_el.get(f'{{{R_NS}}}id')
                    if rid and rid in rid_to_target:
                        xml_to_sheetname[rid_to_target[rid]] = sname

            print(f"[break_external_links] Sheet mapping ({len(xml_to_sheetname)} sheets):")
            for xml_path, sname in xml_to_sheetname.items():
                print(f"    {xml_path} → '{sname}'")

            # ── 3. Rewrite each entry in the zip ───────────────────────
            for item in zin.infolist():
                name = item.filename

                if name in ext_link_parts or name in ext_link_rels:
                    print(f"[break_external_links] Removing part: {name}")
                    continue

                data = zin.read(name)

                try:
                    # 3a. workbook.xml — strip <externalReferences> and ext-link definedNames
                    if name.lower().endswith('workbook.xml') and '/_rels/' not in name.lower():
                        root = _parse(data)
                        for ext_refs in root.findall(f'{{{NS}}}externalReferences'):
                            root.remove(ext_refs)
                        for defined_names in root.findall(f'{{{NS}}}definedNames'):
                            for dn in list(defined_names):
                                val = dn.text or ''
                                if '[' in val:
                                    defined_names.remove(dn)
                        data = _serialize(root)

                    # 3b. workbook.xml.rels — remove externalLink relationships
                    elif wb_rels_name and name == wb_rels_name:
                        root = _parse(data)
                        for rel in list(root):
                            if rel.get('Type') == EXT_LINK_TYPE:
                                root.remove(rel)
                        data = _serialize(root)

                    # 3c. [Content_Types].xml — remove ext link content types
                    elif name == '[Content_Types].xml':
                        root = _parse(data)
                        for node in list(root):
                            ct = node.get('ContentType', '')
                            part_name = node.get('PartName', '').lstrip('/')
                            if ct == EXT_LINK_CT:
                                root.remove(node)
                            elif part_name in ext_link_parts or part_name in ext_link_rels:
                                root.remove(node)
                        data = _serialize(root)

                    # 3d. Worksheet XML — remove external link formulas only
                    elif '/worksheets/' in name and name.endswith('.xml'):
                        root = _parse(data)
                        changed = False
                        _ext_link_re = re.compile(r'\[\d+\]')

                        for c in root.iter(f'{{{NS}}}c'):
                            f_el = c.find(f'{{{NS}}}f')
                            if f_el is None:
                                continue

                            formula_text = f_el.text or ''
                            v_el = c.find(f'{{{NS}}}v')
                            is_ext_link = _ext_link_re.search(formula_text)

                            if is_ext_link:
                                # External link formula: remove formula, keep cached value
                                c.remove(f_el)
                                if v_el is None:
                                    new_v = ET.SubElement(c, f'{{{NS}}}v')
                                    new_v.text = '0'
                                changed = True

                        if changed:
                            data = _serialize(root)

                except Exception as xml_err:
                    print(f"[break_external_links] Could not process {name}: {xml_err}")
                    data = zin.read(name)

                zout.writestr(item, data)

        shutil.move(tmp_path, str(path))
        print(f"[break_external_links] External links removed from: {path.name}")

    except Exception as e:
        print(f"[break_external_links] Failed: {e}")
        traceback.print_exc()
        if tmp_path:
            try:
                if Path(tmp_path).exists():
                    os.remove(tmp_path)
            except Exception:
                pass

def load_mappings_from_json() -> List[Dict[str, Any]]:

        return []

        with open(MAPPINGS_FILE, "r", encoding="utf-8") as f:

def get_mappings_for_month_year(month_name: str, year: int) -> List[Dict]:
    """FINAL FIXED VERSION - always forces clean 'REFI CM' (no 'Worksheet ', no year)"""
    filepath = MAPPINGS_BUDGET_REFI_PREV_YEAR
    month_lower = month_name.lower().strip()

    print(f"[DEBUG] Looking for mappings for: {month_name} {year}")

    if not filepath.is_file():
        print(f"[DEBUG] ERROR: {filepath.name}  !")
        return []

    try:
        with open(filepath, "r", encoding="utf-8-sig") as f:
            data = json.load(f)
            if isinstance(data, dict):
                data = [data]

        if not isinstance(data, list):
            print(f"[DEBUG] ERROR: {filepath.name} has unexpected format")
            return []

        matched = next(
            (entry for entry in data if str(entry.get("month", "")).lower().strip() == month_lower),
            None
        )
        if not matched:
            print(f"[DEBUG] No mappings found for {month_name} in {filepath.name}")
            return []

        resolved = []
        for src in matched.get("sources", []):
            template = src.get("source_sheet_template", "").strip()

            # === FORCE CLEAN SHEET NAMES ===
            if "REFI CM" in template.upper():
                resolved_sheet = "REFI CM"  #always clean
            elif "{year-1}" in template:
                resolved_sheet = template.replace("{year-1}", str(year - 1))
            elif "{year}" in template:
                # Budget sheets stay in selected year, plain year uses previous year
                if "budget" in template.lower():
                    resolved_sheet = template.replace("{year}", str(year))
                else:
                    resolved_sheet = template.replace("{year}", str(year - 1))
            else:
                resolved_sheet = template

            # Extra safety: remove any "Worksheet " prefix that might be in JSON
            resolved_sheet = resolved_sheet.replace("Worksheet ", "").strip()

            row_map = src.get("rowMap", {})
            source_rows = src.get("source_rows", [])
            dest_rows = src.get("dest_rows", [])

            # Support rowMap (dest -> source)
            if isinstance(row_map, dict) and row_map:
                source_rows = []
                dest_rows = []
                for dest_row_str, src_row in row_map.items():
                    if src_row == 0 or src_row is None:
                        continue
                    dest_rows.append(int(dest_row_str))
                    source_rows.append(int(src_row))

            if row_map and not source_rows:
                print(f"[WARN] rowMap provided but produced no row pairs for sheet {resolved_sheet}")
            resolved.append({
                "source_sheet": resolved_sheet,
                "source_column": src.get("source_column", "N").upper(),
                "source_rows": source_rows,
                "dest_sheet": src.get("dest_sheet", "MZLZ Consolidated"),
                "dest_column": src.get("dest_column", "G").upper(),
                "dest_rows": dest_rows,
                "rowMap": row_map if isinstance(row_map, dict) else {}
            })
            print(f"[DEBUG] Added source: '{resolved_sheet}'{src.get('dest_column')}")

        print(f"[DEBUG] Total sources created: {len(resolved)}")
        return resolved

    except Exception as e:
        print(f"[DEBUG] ERROR Loading JSON: {e}")
        traceback.print_exc()
        return []

class PeriodRow(QFrame):
    removed = pyqtSignal(object)
    monthsChanged = pyqtSignal()
    monthSelected = pyqtSignal(int, str)

    # CHANGE THIS NUMBER TO MAKE THE CARD NARROWER
    CARD_WIDTH = 450  # Try 400, 420, 430, 450, 460... smaller = narrower

    def __init__(self, year: int, parent=None):
        super().__init__(parent)
        self.year = year
        self.month_checkboxes = {}
        self.setMaximumWidth(PeriodRow.CARD_WIDTH)  # forces the narrow width
        self._build_ui()

            cb.stateChanged.connect(lambda state, m=month: self._on_month_checkbox_changed(m, state))

    def _on_month_checkbox_changed(self, month: str, state):

        quick = QuarterQuickSelector(self)
        layout.addWidget(quick)

        # Super-tight month grid
        months_grid = QGridLayout()
        months_grid.setSpacing(3)  # very tight

        return [m for m, cb in self.month_checkboxes.items() if cb.isChecked()]

    def is_empty(self) -> bool:
        return not any(cb.isChecked() for cb in self.month_checkboxes.values())

                    self.dest_rows.append(int(dest))

                    pass

        if self.on_change:
            self.on_change()

    def _on_selected_changed(self, state):
        if self._sync_peer and not self._is_syncing:
            try:
                self._sync_peer._is_syncing = True
                try:
                    self._sync_peer.checkbox.setChecked(self.is_selected())
                except Exception:
                    self._sync_peer = None
            finally:
                if self._sync_peer:
                    self._sync_peer._is_syncing = False
        self._on_field_change()

    def _bind_sync_peer(self, peer):
        if self._sync_peer:
            try:
                self._sync_peer.destroyed.disconnect(self._clear_sync_peer)
            except Exception:
                pass
        self._sync_peer = peer
        if peer:
            try:
                peer.destroyed.connect(self._clear_sync_peer)
            except Exception:
                pass

    def _clear_sync_peer(self):
        self._sync_peer = None

    def _sync_text_field(self, widget, value):
        if not self._sync_peer or self._is_syncing:
            return
        peer_widget = self._find_peer_widget(widget)
        if not peer_widget:
            return
        if peer_widget.text() == value:
            return
        try:
            self._sync_peer._is_syncing = True
            peer_widget.blockSignals(True)
            peer_widget.setText(value)
        except RuntimeError:
            self._sync_peer = None
        finally:
            if peer_widget:
                peer_widget.blockSignals(False)
            if self._sync_peer:
                self._sync_peer._is_syncing = False

    def _on_copy_full_sheet_changed(self, checked):
        if self._sync_peer and not self._is_syncing:
            try:
                peer_widget = self._sync_peer.chk_copy_full_sheet
            except Exception:
                self._sync_peer = None
                peer_widget = None
            if peer_widget is None:
                return
            if peer_widget.isChecked() == checked:
                self._on_field_change()
                return
            try:
                self._sync_peer._is_syncing = True
                peer_widget.blockSignals(True)
                peer_widget.setChecked(checked)
            except Exception:
                self._sync_peer = None
            finally:
                if peer_widget:
                    peer_widget.blockSignals(False)
                if self._sync_peer:
                    self._sync_peer._is_syncing = False
        self._on_field_change()

    def _on_custom_sheet_name_changed(self, text):
        if self._sync_peer and not self._is_syncing:
            peer_widget = self._sync_peer.custom_sheet_name
            if peer_widget is None:
                return
            if peer_widget.text() == text:
                self._on_field_change()
                return
            try:
                self._sync_peer._is_syncing = True
                peer_widget.blockSignals(True)
                peer_widget.setText(text)
            finally:
                peer_widget.blockSignals(False)
                self._sync_peer._is_syncing = False
        self._on_field_change()

    def _find_peer_widget(self, widget):
        if widget is self.src_sheet:
            return self._sync_peer.src_sheet
        if widget is self.src_column:
            return self._sync_peer.src_column
        if widget is self.dest_sheet:
            return self._sync_peer.dest_sheet
        if widget is self.dest_column:
            return self._sync_peer.dest_column
        return None

    def is_selected(self) -> bool:
        """Return True if the checkbox on this card is checked"""
        return self.checkbox.isChecked()

    def set_selected(self, checked: bool):
        """Used by 'Select All' checkbox"""
        self.checkbox.blockSignals(True)
        self.checkbox.setChecked(checked)
        self.checkbox.blockSignals(False)

            'copy_full_sheet': self.chk_copy_full_sheet.isChecked(),

        }

    def get_values(self) -> dict:
        """Compatibility method used by MappingPopup + execute_single + auto-popup"""
        source_file_type = getattr(self, 'source_file_type', 'cost_control')
        row_map = {}
        if source_file_type == 'sap_ytd' and self.mappings_file:
            try:
                import json
                import re
                raw_text = ""
                with open(self.mappings_file, 'r', encoding='utf-8') as f:
                    raw_text = f.read()
                # Detect duplicate keys in rowMap blocks
                rowmap_blocks = re.findall(r'"rowMap"\s*:\s*\{(.*?)\}', raw_text, flags=re.S)
                for block in rowmap_blocks:
                    keys = re.findall(r'"(\d+)"\s*:', block)
                    dupes = sorted({k for k in keys if keys.count(k) > 1})
                    if dupes:
                        print(f"[WARN] Duplicate rowMap keys in JSON: {', '.join(dupes)}")
                all_entries = json.loads(raw_text)
                match = next((e for e in all_entries if e.get('month') == self.month_name), None)
                print(f"[DEBUG get_values] month_name={self.month_name} year={self.year} match={'yes' if match else 'no'}")
                if match:
                    row_map = match.get('rowMap', {})
            except Exception as e:
                print(f"[DEBUG get_values] Failed to load rowMap: {e}")

        return {
            'index': getattr(self, 'index', 0),
            'month': self.month_name,
            'year': self.year,
            'source_sheet': self.src_sheet.text().strip(),
            'source_column': self.src_column.text().strip().upper(),
            'source_rows': list(self.source_rows_data),  # safe copy
            'dest_sheet': self.dest_sheet.text().strip(),
            'dest_column': self.dest_column.text().strip().upper(),
            'dest_rows': list(self.dest_rows_data),
            'names': list(self.names_data),
            'source_file_type': source_file_type,
            'selected': self.is_selected(),
            'mappings_file': str(self.mappings_file) if self.mappings_file else None,
            'mapping_label': self.mappings_label.text() if hasattr(self, 'mappings_label') else "",
            'copy_full_sheet': self.chk_copy_full_sheet.isChecked(),
            'custom_sheet_name': self.custom_sheet_name.text().strip() or MONTH_TO_NUM.get(self.month_name, "01"),
            'rowMap': row_map,
        }


#  BACKGROUND LOAD WORKER
class TransferWorker(QObject):
    """Runs execute_all transfers in a background thread with pause/stop support."""
    progress = pyqtSignal(int, int, str)  # (current, total, message)
    finished = pyqtSignal(int, int, list)  # (total_cells, executed, skipped)
    error = pyqtSignal(str)
    row_done = pyqtSignal(int, bool, str)  # (row_index, success, message)

    def __init__(self, window, rows):
        super().__init__()
        self._window = window
        self._rows = rows
        self._stop_requested = False
        self._pause_requested = False
        self._pause_lock = __import__('threading').Event()
        self._pause_lock.set()  # start unpaused

    def stop(self):
        self._stop_requested = True
        self._pause_lock.set()  # unblock if paused so thread can exit

    def pause(self):
        self._pause_requested = True
        self._pause_lock.clear()

    def resume(self):
        self._pause_requested = False
        self._pause_lock.set()

    def run(self):
        total = len(self._rows)
        total_cells = 0
        executed = 0
        skipped = []

        for idx, row in enumerate(self._rows):
            # Check stop
            if self._stop_requested:
                break

            # Check pause (blocks here until resumed or stopped)
            self._pause_lock.wait()
            if self._stop_requested:
                break

            m = row.get_values()
            read_type = m.get('source_file_type', 'cost_control')
            read_key = (m.get('month'), m.get('year'), read_type)
            write_key = (m.get('month'), m.get('year'), 'cost_control')
            legacy_key = (m.get('month'), m.get('year'))

            self.progress.emit(idx + 1, total,
                               f"Running {m.get('month', '')} {m.get('year', '')}    card {idx + 1}/{total}")

            can_execute = True
            if read_key not in self._window.loaded_workbooks:
                if not (legacy_key in self._window.loaded_workbooks and read_type == 'cost_control'):
                    can_execute = False
                    skipped.append(f"{read_type} ({m.get('month')} {m.get('year')})")
            if write_key not in self._window.loaded_workbooks and \
                    legacy_key not in self._window.loaded_workbooks:
                can_execute = False
                if f"cost_control ({m.get('month')} {m.get('year')})" not in skipped:
                    skipped.append(f"cost_control ({m.get('month')} {m.get('year')})")

            row_map = m.get('rowMap', {}) if isinstance(m, dict) else {}
            is_sap_ytd = m.get('source_file_type') == 'sap_ytd'

            if not can_execute:
                self.row_done.emit(idx, False, "Skipped")
                continue

            if is_sap_ytd:
                if not row_map:
                    self.row_done.emit(idx, False, "Skipped")
                    continue
            else:
                if not m['source_rows'] or not m['dest_rows']:
                    self.row_done.emit(idx, False, "Skipped")
                    continue

            try:
                cells = self._window._transfer(m, deferred_save=True)
                total_cells += cells
                executed += 1
                self.row_done.emit(idx, True, f"{cells} cells")
            except Exception as e:
                self.row_done.emit(idx, False, str(e))
                print(f"[TransferWorker] Transfer error: {e}")
                traceback.print_exc()

        #   DEBUG: Show all keys in loaded_workbooks
        print(f"[DEBUG] All loaded_workbooks keys: {list(self._window.loaded_workbooks.keys())}")
        cost_control_keys = [k for k in self._window.loaded_workbooks.keys() if len(k) == 3 and k[2] == "cost_control"]
        print(f"[DEBUG] Keys matching cost_control filter: {cost_control_keys} (count: {len(cost_control_keys)})")

        #   Batch save + finalize once per destination file
        saved_paths = set()
        for key, info in self._window.loaded_workbooks.items():
            if len(key) == 3 and key[2] == "cost_control":
                wb = info.get("workbook")
                path = info.get("path")
                if wb is None or path is None or path in saved_paths:
                    continue
                try:
                    wb.calculation.calcMode = "manual"
                    print(f"[Batch] Saving {Path(path).name}")
                    wb.save(path)
                    wb.calculation.calcMode = "auto"
                    _wbv = info.get("wb_values")
                    break_external_links(path, wb_values=_wbv)
                    self._window._finalize_with_excel_calc(path)
                    saved_paths.add(path)
                except Exception as _se:
                    print(f"[Batch] Save error for {path}: {_se}")

        self.finished.emit(total_cells, executed, skipped)


class LoadWorker(QObject):
    progress = pyqtSignal(int, str)  # (step, message)
    result = pyqtSignal(dict)  # {'loaded': {}, 'missing': [], 'errors': []}
    error = pyqtSignal(str)  # error message
    finished = pyqtSignal()

    def __init__(self, selected_periods, source_folder, dest_folder):
        super().__init__()
        self.selected_periods = selected_periods
        self.source_folder = Path(source_folder)
        self.dest_folder = Path(dest_folder)

    def run(self):
        loaded = {}
        missing = []
        errors = []

        # Track what we've already processed in this run to avoid duplicates
        processed_periods = set()

        try:
            total = max(len(self.selected_periods), 1)

            for idx, (month_name, year) in enumerate(sorted(self.selected_periods), start=1):
                # Skip duplicates in selected_periods
                period_key = (month_name, year)
                if period_key in processed_periods:
                    print(f"[DEBUG] Skipping duplicate: {month_name} {year}")
                    continue
                processed_periods.add(period_key)

                month_num = MONTH_TO_NUM.get(month_name, "01")
                month_num_plain = str(int(month_num))  # "02" -> "2"
                month_dir = month_num  # Keep zero-padded: "02", not "2"
                self.progress.emit(idx, f"Loading {month_name} {year}...")

                # Construct paths for Cost Control and SAP files
                dest_path = Path(self.dest_folder)

                # Check if dest_folder is already pointing to a specific month folder
                # e.g., L:\...\2026\02\test or L:\...\2026\02
                path_parts = dest_path.parts

                # Find year and month from the path itself
                found_year = None
                found_month = None
                for part in path_parts:
                    if part.isdigit() and len(part) == 4:  # Year like 2026
                        found_year = part
                    elif part.isdigit() and len(part) == 2:  # Month like 02
                        found_month = part

                if found_year and found_month:
                    # User's folder already has year/month - use it directly
                    # For the selected month/year, look in that same parent structure
                    base_folder = dest_path.parent.parent  # Go up to year folder level
                    cost_control_path = base_folder / month_num / "test"
                    sap_path = base_folder / month_num / "SAP export monthly"
                else:
                    # No year/month in path - construct full path
                    base_path = dest_path / str(year) / month_dir
                    cost_control_path = base_path / "test"
                    sap_path = base_path / "SAP export monthly"

                print(f"[DEBUG] Cost Control path: {cost_control_path}")
                print(f"[DEBUG] SAP path: {sap_path}")

                # === COST CONTROL FILE ===
                cc_file = None
                if cost_control_path.exists() and cost_control_path.is_dir():
                    # Debug: List all files in the directory
                    all_files = list(cost_control_path.iterdir())
                    print(f"[DEBUG] Cost Control files ({len(all_files)} total):")
                    for f in all_files:
                        if f.is_file():
                            print(f"[DEBUG]   - {f.name}")

                    for filename in cost_control_path.iterdir():
                        if filename.is_file():
                            name_lower = filename.name.lower()
                            has_excel_ext = filename.suffix.lower() in (".xlsx", ".xlsm", ".xls")

                            # Match: Cost Control ZAG {month}_{year}_working.{ext}
                            expected_base = f"cost control zag {month_num}_{year}_working"
                            expected_base_plain = f"cost control zag {month_num_plain}_{year}_working"
                            base_name = filename.stem.lower()

                            if has_excel_ext and (expected_base in name_lower or expected_base_plain in name_lower or
                                                  expected_base == base_name or expected_base_plain == base_name):
                                cc_file = filename
                                print(f"[DEBUG] Cost Control MATCH: {filename.name}")
                                break
                else:
                    print(f"[DEBUG] Cost Control path does not exist: {cost_control_path}")

                if cc_file:
                    try:
                        # Load editable workbook
                        wb, used_xlwings = load_excel_with_fallback(str(cc_file), keep_vba=True)
                        #   CRITICAL: load data_only snapshot of the DESTINATION file
                        # This captures pre-modification cached formula values so that
                        # break_external_links can re-inject them after openpyxl strips them on save.
                        try:
                            wb_values_dest = openpyxl.load_workbook(str(cc_file), data_only=True, keep_vba=True)
                        except Exception as e_donly:
                            print(f"[DEBUG] Could not load data_only for Cost Control: {e_donly}")
                            wb_values_dest = None
                        loaded[(month_name, year, "cost_control")] = {
                            'workbook': wb,
                            'wb_values': wb_values_dest,  #  destination snapshot (pre-modification)
                            'path': str(cc_file),
                            'type': 'cost_control',
                            'tmp_path': None
                        }
                        print(f"[DEBUG] Cost Control loaded: {cc_file.name}")
                        self.progress.emit(idx, f"Cost Control loaded: {cc_file.name}")
                    except Exception as e:
                        errors.append((month_name, year, f"Cost Control: {e}"))
                        print(f"[DEBUG] Failed to load Cost Control: {e}")
                        self.progress.emit(idx, f"Failed Cost Control: {e}")
                else:
                    missing.append((month_name, year, f"Cost Control ZAG {month_num}_{year}_working.xlsm"))
                    print(f"[DEBUG] Cost Control file   in {cost_control_path}")

                # === SAP FILE ===
                sap_file = None
                if sap_path.exists() and sap_path.is_dir():
                    # Debug: List all files in the SAP directory
                    all_sap_files = list(sap_path.iterdir())
                    print(f"[DEBUG] SAP files ({len(all_sap_files)} total):")
                    for f in all_sap_files:
                        if f.is_file():
                            print(f"[DEBUG]   - {f.name}")

                    for filename in sap_path.iterdir():
                        if filename.is_file():
                            name_lower = filename.name.lower()
                            file_stem = filename.stem.lower()
                            has_excel_ext = filename.suffix.lower() in (".xlsx", ".xlsm", ".xls")

                            # Match multiple patterns:
                            # 1. SAP export monthly.{ext}
                            # 2. {month}.{ext} (e.g., 2.xlsx)
                            # 3. {month}_{year}.{ext} (e.g., 02_2026.xlsx)
                            is_sap_export = "sap" in name_lower and "export" in name_lower and "monthly" in name_lower
                            is_month_file = file_stem == month_num_plain or file_stem == month_num
                            is_month_year_file = file_stem == f"{month_num}_{year}" or file_stem == f"{month_num_plain}_{year}"

                            if has_excel_ext and (is_sap_export or is_month_file or is_month_year_file):
                                sap_file = filename
                                print(f"[DEBUG] SAP MATCH: {filename.name}")
                                break
                else:
                    print(f"[DEBUG] SAP path does not exist: {sap_path}")

                if sap_file and cc_file:
                    try:
                        wb, used_xlwings = load_excel_with_fallback(str(sap_file), data_only=True, read_only=True)
                        loaded[(month_name, year, "sap")] = {
                            'workbook': wb,
                            'path': str(sap_file),
                            'type': 'sap'
                        }
                        print(f"[DEBUG] SAP loaded: {sap_file.name}")
                        self.progress.emit(idx, f"SAP loaded: {sap_file.name}")
                    except Exception as e:
                        errors.append((month_name, year, f"SAP: {e}"))
                        print(f"[DEBUG]Failed to load SAP: {e}")
                        self.progress.emit(idx, f"Failed SAP: {e}")

                # === AUTO-LOAD SAP YTD FILE ===
                # YTD_{MM}_{YYYY}.xlsx in SAP export monthly folder   sheet "Report"   column C
                month_num_padded_str = month_num.zfill(2) if month_num else ''
                ytd_key = (month_name, year, 'sap_ytd')
                if ytd_key not in loaded:
                    ytd_file = None
                    if sap_path.exists() and sap_path.is_dir():
                        for filename in sap_path.iterdir():
                            if filename.is_file():
                                stem = filename.stem.upper()
                                has_excel_ext = filename.suffix.lower() in (".xlsx", ".xlsm", ".xls")
                                is_ytd = stem.startswith("YTD_") and has_excel_ext
                                if is_ytd:
                                    ytd_file = filename
                                    break
                    if ytd_file:
                        try:
                            ytd_wb, _ = load_excel_with_fallback(str(ytd_file), data_only=True, read_only=True)
                            loaded[ytd_key] = {
                                'workbook': ytd_wb,
                                'path': str(ytd_file),
                                'type': 'sap_ytd'
                            }
                            print(f"[DEBUG] SAP YTD loaded: {ytd_file.name} for {month_name} {year}")
                        except Exception as e:
                            print(f"[DEBUG] Could not load SAP YTD for {month_name} {year}: {e}")
                    else:
                        print(f"[DEBUG] SAP YTD file (YTD_*.xlsx)   in {sap_path}")

                # (No separate prev-month SAP file needed    the current SAP monthly export
                #  contains ALL month sheets ("1".."12"). Formula rows 118-123 and 142-147
                #  reference the previous month's sheet number within the same file.
                #  e.g. for June   sheet '5' (May) is already in 06_2025.xlsx or 6.xlsx)
                elif sap_file and not cc_file:
                    missing.append((month_name, year, f"Cost Control ZAG {month_num}_{year}_working.xlsm"))
                    print(f"[DEBUG] Cost Control  , skipping SAP")
                elif not sap_file and cc_file:
                    missing.append((month_name, year, f"SAP export monthly.xlsx"))
                    print(f"[DEBUG] SAP file   in {sap_path}")

                # === STAFF FILE ===
                staff_file = None
                # Staff file path: L:\Cost control\Cost Control\Cost control\Staff\Staff_{year}.xlsx
                base_staff_path = Path("L:/Cost control/Cost Control/Cost control/Staff")
                staff_filename = f"Staff_{year}.xlsx"
                staff_path = base_staff_path / staff_filename

                if staff_path.exists():
                    print(f"[DEBUG]Staff file found: {staff_path}")
                    staff_file = staff_path
                else:
                    print(f"[DEBUG] Staff file  : {staff_path}")

                if staff_file and cc_file:
                    try:
                        wb, used_xlwings = load_excel_with_fallback(str(staff_file), data_only=True, read_only=True)
                        loaded[(month_name, year, "staff")] = {
                            'workbook': wb,
                            'path': str(staff_file),
                            'type': 'staff'
                        }
                        print(f"[DEBUG] Staff loaded: {staff_file.name}")
                        self.progress.emit(idx, f"Staff loaded: {staff_file.name}")
                    except Exception as e:
                        errors.append((month_name, year, f"Staff: {e}"))
                        print(f"[DEBUG] Failed to load Staff: {e}")
                        self.progress.emit(idx, "Failed Staff: {e}")

                # === PAX FILE ===
                pax_file = None
                base_pax_path = Path("L:/Cost control/Cost Control/Cost control")
                traffic_dir = base_pax_path / str(year) / month_dir / "Traffic"
                pax_stem = f"TRAFFIC mott {year}"
                pax_exact_candidates = list(traffic_dir.glob(f"{pax_stem}*.xls*")) if traffic_dir.exists() else []
                pax_path = traffic_dir / f"{pax_stem}.xlsx"

                if pax_path.exists():
                    pax_file = pax_path
                elif pax_exact_candidates:
                    pax_file = pax_exact_candidates[0]

                if pax_file:
                    print(f"[DEBUG] PAX file found: {pax_file}")
                else:
                    print(f"[DEBUG] PAX file   in {traffic_dir} (expected {pax_stem}*.xls*)")

                if pax_file and cc_file:
                    try:
                        wb, used_xlwings = load_excel_with_fallback(str(pax_file), data_only=True, read_only=True)
                        loaded[(month_name, year, "pax")] = {
                            'workbook': wb,
                            'path': str(pax_file),
                            'type': 'pax'
                        }
                        print(f"[DEBUG] PAX loaded: {pax_file.name}")
                        self.progress.emit(idx, f"PAX loaded: {pax_file.name}")
                    except Exception as e:
                        errors.append((month_name, year, f"PAX: {e}"))
                        print(f"[DEBUG] Failed to load PAX: {e}")
                        self.progress.emit(idx, f"Failed PAX: {e}")

            self.result.emit({'loaded': loaded, 'missing': missing, 'errors': errors})
        except Exception as e:
            self.error.emit(str(e))
        finally:
            self.finished.emit()

    def _get_default_source_sheet(self, month_name: str, year: int) -> str:
        """Case-insensitive fuzzy matching + clean sheet names"""
        key = (month_name, year)
        if key not in self.loaded_workbooks:
            return "Report"

        wb = self.loaded_workbooks[key]['workbook']
        terms = [f"refi cm", str(year), f"{year} budget", month_name.lower()]

        for sheet in wb.sheetnames:
            clean_sheet = sheet.replace("Worksheet ", "").strip()
            s = clean_sheet.lower()
            if any(t in s for t in terms):
                return clean_sheet  # return clean name

        return wb.sheetnames[0] if wb.sheetnames else "Report"

    def _create_cost_control_files(self):
        selected_periods = []
        for row in self._get_all_period_rows():
            year = row.year
            for month_name, cb in row.month_checkboxes.items():
                if cb.isChecked():
                    selected_periods.append((month_name, year))

        if not selected_periods:
            QMessageBox.warning(self, "No Selection", "Please select at least one month.")
            return

        errors = []
        created = []
        skipped = []

        for month_name, year in selected_periods:
            try:
                result = self._create_cost_control_for_period(month_name, year)
                status, fallback = result
                if status == "created":
                    created.append((month_name, year))
                    if fallback:
                        fallback_month, fallback_year = fallback
                        errors.append((month_name, year,
                                       f"Used fallback source: {fallback_month} {fallback_year}"))
                elif status == "exists":
                    skipped.append((month_name, year))
                elif status == "skipped":
                    skipped.append((month_name, year))
                    if fallback:
                        errors.append((month_name, year, fallback))
            except Exception as exc:
                errors.append((month_name, year, str(exc)))

        message_lines = []
        if created:
            created_list = ", ".join(f"{m} {y}" for m, y in created)
            message_lines.append(f"Created: {created_list}")
        if skipped:
            skipped_list = ", ".join(f"{m} {y}" for m, y in skipped)
            message_lines.append(f"Already exists: {skipped_list}")
        if errors:
            errors_list = "\n".join(f"{m} {y}: {err}" for m, y, err in errors)
            message_lines.append(f"Errors:\n{errors_list}")

        if not message_lines:
            message_lines.append("No files were created.")

        QMessageBox.information(self, "Cost Control Creation", "\n\n".join(message_lines))
        if created:
            self._update_load_button_state()

    def _export_selected_months(self):
        selected_periods = []
        for row in self._get_all_period_rows():
            year = row.year
            for month_name, cb in row.month_checkboxes.items():
                if cb.isChecked():
                    selected_periods.append((month_name, year))

            QMessageBox.warning(self, "No Selection", "Please select at least one month.")
            return

        if not self.loaded_workbooks:
            QMessageBox.warning(self, "Not Loaded", "Load file(s) first!")
            return

        for month_name, year in selected_periods:

    def _export_combined_for_year(self, year: int, months: List[str]) -> str:
        month_nums = [int(MONTH_TO_NUM.get(month, "01")) for month in months]
        latest_month_num = max(month_nums)
        latest_month_str = f"{latest_month_num:02d}"

        num_to_name = {int(v): k for k, v in MONTH_TO_NUM.items()}

        def _ensure_month_file(month_name: str) -> Path:
            month_num = int(MONTH_TO_NUM.get(month_name, "01"))
            month_str = f"{month_num:02d}"
            month_dir = month_str
            target_dir = self.dest_folder / str(year) / month_dir / "test"
            target_dir.mkdir(parents=True, exist_ok=True)

            target_name = f"Cost Control ZAG {month_str}_{year}_working"
            target_path = self._find_existing_cost_control(target_dir, target_name)
            if not target_path:
                target_path = target_dir / f"{target_name}.xlsm"

            if not target_path.exists():
                # find nearest previous month file (including ones created earlier in this run)
                prev_path = None
                for prev_num in range(month_num - 1, 0, -1):
                    prev_str = f"{prev_num:02d}"
                    prev_dir = self.dest_folder / str(year) / prev_str / "test"
                    prev_name = f"Cost Control ZAG {prev_str}_{year}_working"
                    prev_path = self._find_existing_cost_control(prev_dir, prev_name)
                    if not prev_path:
                        prev_path = prev_dir / f"{prev_name}.xlsm"
                    if prev_path.exists():
                        break
                    prev_path = None

                if not prev_path or not prev_path.exists():
                    raise FileNotFoundError(
                        f"Target file and previous month file missing: {target_path}"
                    )
                shutil.copy2(prev_path, target_path)

            ensure_file_closed(target_path)

            # ensure loaded_workbooks entry exists for this month
            key = (month_name, year, "cost_control")
            if key not in self.loaded_workbooks or self.loaded_workbooks.get(key, {}).get("path") != str(target_path):
                wb = openpyxl.load_workbook(str(target_path), keep_vba=True)
                wb_values = openpyxl.load_workbook(str(target_path), data_only=True, keep_vba=True)
                self.loaded_workbooks[key] = {
                    "path": str(target_path),
                    "workbook": wb,
                    "wb_values": wb_values
                }
            if hasattr(self, "loaded_periods"):
                self.loaded_periods.add((month_name, year))

            return target_path

        # Ensure each selected month exists in order (June -> July -> August)
        for m_num in sorted(month_nums):
            m_name = num_to_name.get(int(m_num), MONTHS_LIST[int(m_num) - 1])
            _ensure_month_file(m_name)

        # target path is the latest month after creation
        target_path = _ensure_month_file(num_to_name.get(int(latest_month_num), MONTHS_LIST[int(latest_month_num) - 1]))

        # Filter out months that are not loaded
        available_months = [month_name for month_name in months if (month_name, year) in self.loaded_periods]
        missing_months = [month_name for month_name in months if (month_name, year) not in self.loaded_periods]
        if missing_months:
            print(f"[WARN] Skipping not-loaded months for {year}: {', '.join(missing_months)}")
        if not available_months:
            raise ValueError(f"No loaded months selected for {year}")

        # Run transfers for selected months only
        selected_rows = []
        for month_name in available_months:
            month_selected = [
                row for row in self.mapping_rows
                if row.is_selected()
                   and row.get_values().get("year") == year
                   and row.get_values().get("month") == month_name
            ]
            if month_selected:
                selected_rows.extend(month_selected)
                continue

            month_rows = [
                row for row in self.mapping_rows
                if row.get_values().get("year") == year
                   and row.get_values().get("month") == month_name
            ]
            if month_rows:
                print(
                    f"[DEBUG] No mapping cards selected for {month_name} {year}; "
                    f"using all {len(month_rows)} rows."
                )
                selected_rows.extend(month_rows)

        if not selected_rows:
            raise ValueError(f"No mapping rows available for {year}")

        latest_month_name = next(
            (name for name in available_months if int(MONTH_TO_NUM.get(name, "01")) == latest_month_num),
            available_months[0]
        )
        latest_key = (latest_month_name, year, "cost_control")
        if latest_key not in self.loaded_workbooks:
            # If latest month workbook was just created, load it now
            if target_path and Path(target_path).exists():
                print(f"[DEBUG] Loading latest month workbook: {target_path}")
                new_wb = openpyxl.load_workbook(str(target_path), keep_vba=True)
                new_wb_values = openpyxl.load_workbook(str(target_path), data_only=True, keep_vba=True)
                self.loaded_workbooks[latest_key] = {
                    "path": str(target_path),
                    "workbook": new_wb,
                    "wb_values": new_wb_values
                }
            else:
                raise ValueError(f"Latest month workbook not loaded: {latest_month_name} {year}")

        latest_wb_info = self.loaded_workbooks[latest_key]
        latest_wb = latest_wb_info["workbook"]

        # If target_path differs from loaded workbook path, we need to update the reference
        # This happens when the latest month file was copied from previous month
        if latest_wb_info["path"] != str(target_path):
            print(f"[DEBUG] ReLoading latest month workbook: {target_path}")
            # Close old workbook to free resources
            try:
                latest_wb_info["workbook"].close()
            except:
                pass
            # Load the new target file
            new_wb = openpyxl.load_workbook(str(target_path), keep_vba=True)
            new_wb_values = openpyxl.load_workbook(str(target_path), data_only=True, keep_vba=True)
            # Update the loaded_workbooks entry
            self.loaded_workbooks[latest_key] = {
                "path": str(target_path),
                "workbook": new_wb,
                "wb_values": new_wb_values
            }
            latest_wb = new_wb

        total_cells = 0
        ytd_row_for_latest = None  # holds the YTD mapping row for the latest month only

        for row in selected_rows:
            mappings = row.get_values()
            mappings["write_month"] = latest_month_name
            mappings["write_year"] = year
            source_file_type = mappings.get("source_file_type", "cost_control")
            month_name = mappings.get("month")

            # YTD cumulative card: only run for the latest selected month, after all others
            if source_file_type == "ytd":
                if month_name == latest_month_name:
                    ytd_row_for_latest = mappings  # save for after all transfers
                continue  # skip YTD cards during normal loop

            total_cells += self._transfer(mappings, deferred_save=True)

        # Run YTD cards only up to the month BEFORE the latest
        # Example: June+July+August   run IU (June) + IV (July), skip IW (August)
        MONTH_ORDER = ['January','February','March','April','May','June',
                       'July','August','September','October','November','December']
        ytd_rows_by_month = {}
        for row in selected_rows:
            mappings = row.get_values()
            if mappings.get('source_file_type') == 'ytd':
                ytd_rows_by_month[mappings.get('month')] = mappings

        for month in MONTH_ORDER:
            if month in ytd_rows_by_month and month != latest_month_name:
                m_mappings = ytd_rows_by_month[month]
                m_mappings['write_month'] = latest_month_name
                m_mappings['write_year'] = year
                print(f"[DEBUG EXPORT] Running YTD card for: {month}")
                total_cells += self._transfer(m_mappings, deferred_save=True)

        print(f"[DEBUG] Saving combined export workbook to: {target_path}")
        latest_wb.save(str(target_path))
        break_external_links(str(target_path), wb_values=self.loaded_workbooks.get(latest_key, {}).get("wb_values"))
        self._finalize_with_excel_calc(str(target_path))

        return str(target_path)

    def _create_cost_control_for_period(self, month_name: str, year: int) -> tuple:
        month_num = MONTH_TO_NUM.get(month_name, "01")
        month_num_plain = str(int(month_num))

        base_dir = self.dest_folder / str(year) / month_num
        test_dir = base_dir / "test"
        target_dir = test_dir if test_dir.exists() else base_dir
        target_dir.mkdir(parents=True, exist_ok=True)

        target_name = f"Cost Control ZAG {month_num}_{year}_working"
        target_path = self._find_existing_cost_control(target_dir, target_name)
        if not target_path:
            target_path = target_dir / f"{target_name}.xlsm"

        ensure_file_closed(target_path)

        if target_path.exists():
            return "exists", None

        start_index = MONTHS_LIST.index(month_name)
        search_year = year
        fallback_info = None
        source_file = None

        # search backwards up to 12 months
        for _ in range(12):
            start_index -= 1
            if start_index < 0:
                start_index = len(MONTHS_LIST) - 1
                search_year -= 1
            candidate_month = MONTHS_LIST[start_index]
            candidate_month_num = MONTH_TO_NUM.get(candidate_month, "01")
            base_source_dir = self.dest_folder / str(search_year) / candidate_month_num
            source_dir = (base_source_dir / "test") if (base_source_dir / "test").exists() else base_source_dir
            if not source_dir.is_dir():
                continue
            source_file = self._find_cost_control_source(source_dir, search_year, candidate_month_num)
            if source_file:
                if candidate_month != MONTHS_LIST[(MONTHS_LIST.index(month_name) - 1) % len(MONTHS_LIST)] or search_year != year:
                    fallback_info = (candidate_month, search_year)
                break

        if not source_file:
            dec_year = year - 1
            dec_month = "December"
            dec_month_num = MONTH_TO_NUM.get(dec_month, "12")
            base_dec_dir = self.dest_folder / str(dec_year) / dec_month_num
            dec_dir = (base_dec_dir / "test") if (base_dec_dir / "test").exists() else base_dec_dir
            dec_file = None
            if dec_dir.is_dir():
                dec_file = self._find_cost_control_source(dec_dir, dec_year, dec_month_num)
            if dec_file:
                response = QMessageBox.question(
                    self,
                    "Fallback source not found",
                    f"No source file found in previous 12 months for {month_name} {year}.\n"
                    f"Use December {dec_year} instead?",
                    QMessageBox.StandardButton.Yes | QMessageBox.StandardButton.No
                )
                if response == QMessageBox.StandardButton.Yes:
                    source_file = dec_file
                    fallback_info = (dec_month, dec_year)
                else:
                    return "skipped", f"User declined December {dec_year}"
            else:
                return "skipped", f"No December {dec_year} file found"

        target_ext = source_file.suffix
        target_path = target_dir / f"{target_name}{target_ext}"
        shutil.copy2(source_file, target_path)
        return "created", fallback_info

    def _find_cost_control_source(self, folder: Path, year: int, month_num: str) -> Optional[Path]:
        month_num_plain = str(int(month_num))
        candidates = [
            f"Cost Control ZAG {month_num}_{year}_working",
            f"Cost Control ZAG {month_num_plain}_{year}_working",
            f"{month_num}_{year}",
            f"{month_num_plain}_{year}",
        ]

        for file in folder.iterdir():
            if not file.is_file():
                continue
            if file.suffix.lower() not in (".xlsx", ".xlsm", ".xls"):
                continue
            stem_lower = file.stem.lower()
            if any(candidate.lower() in stem_lower for candidate in candidates):
                return file
        return None

    def _find_existing_cost_control(self, folder: Path, base_name: str) -> Optional[Path]:
        for ext in (".xlsx", ".xlsm", ".xls"):
            candidate = folder / f"{base_name}{ext}"
            if candidate.exists():
                return candidate
        for file in folder.iterdir():
            if file.is_file() and file.suffix.lower() in (".xlsx", ".xlsm", ".xls"):
                if base_name.lower() in file.stem.lower():
                    return file
        return None

    def _clear_period_rows(self):
        """Safely clear all period rows without crashing"""
        if not hasattr(self, 'periods_layout'):
            return
        # Safety check
        if self.periods_layout.count() == 0:
            self.period_status.setText("Nothing to clear")
            self.period_status.setStyleSheet("color: #6B7280; font-style: italic;")
            return

        # Step 1: Remove all year cards safely (prevents recursion/stack overflow)
        while self.periods_layout.count() > 0:
            item = self.periods_layout.takeAt(0)
            if item and (widget := item.widget()):
                widget.deleteLater()  # safe deletion

        # Step 2: Reset UI state
        self._sync_year_checkbox_with_rows()

        # Step 3: Update status
        self.period_status.setText("All periods cleared")
        self.period_status.setStyleSheet("color: #DC2626; font-style: italic;")
        self._update_load_button_state()

        # Optional: force Qt to refresh the layout
        QTimer.singleShot(10, self.periods_container.updateGeometry)

    def _reset_app_state(self):
        if QMessageBox.question(
                self,
                "Reset",
                "Reset the app state This will clear selections and loaded files.",
                QMessageBox.StandardButton.Yes | QMessageBox.StandardButton.No
        ) != QMessageBox.StandardButton.Yes:
            return

        # Block all signals to prevent cascading events during reset
        self.blockSignals(True)

        # Clear loaded workbooks first (important for file locks)
        for info in list(self.loaded_workbooks.values()):
            try:
                info['workbook'].close()
            except:
                pass
        self.loaded_workbooks.clear()
        self.loaded_periods.clear()

        # Collect all widgets to delete first (avoid deletion during iteration)
        period_widgets = []
        while self.periods_layout.count() > 0:
            item = self.periods_layout.takeAt(0)
            if item and item.widget():
                period_widgets.append(item.widget())

        mapping_widgets = list(self.mapping_rows)
        group_widgets = list(self.mapping_year_groups.values())

        # Delete outside of layout iteration
        for w in period_widgets:
            w.deleteLater()

        for w in mapping_widgets:
            w.deleteLater()
        self.mapping_rows.clear()

        for g in group_widgets:
            g.deleteLater()
        self.mapping_year_groups.clear()

        # Clear file table
        self._update_files_table([], [])

        # Reset year checkboxes
        for year, cb in self.year_checkboxes.items():
            cb.setChecked(False)

        current_year = datetime.now().year
        if current_year in self.year_checkboxes:
            self.year_checkboxes[current_year].setChecked(True)

        self.selected_global_quarters.clear()
        self.mapping_counter = 0

        # Show waiting state
        self._clear_all_mapping_rows()  # This will recreate no_mappings_label

        self.period_status.setText("Select years   Generate   Check months   LOAD")
        self.period_status.setStyleSheet("color:#475569; font-style:italic; font-size:13px;")
        self._show_waiting_state()

        # Unblock signals and update
        self.blockSignals(False)
        self._update_load_button_state()

        Toast(self, "Reset complete - start fresh!")

    def _get_all_period_rows(self):
        """Returns every PeriodRow, even when wrapped in yearCard"""
        rows = []
        for i in range(self.periods_layout.count()):
            item = self.periods_layout.itemAt(i)
            if not item:
                continue
            widget = item.widget()
            if isinstance(widget, PeriodRow):
                rows.append(widget)
            elif isinstance(widget, QFrame) and widget.objectName() == "yearCard":
                # Find the PeriodRow inside the card
                for child in widget.findChildren(PeriodRow):
                    rows.append(child)
                    break
        return rows

    def _remove_period_row(self, period_row):
        year_being_removed = period_row.year

        # Find the parent card and remove it
        card = period_row
        while card and not (isinstance(card, QFrame) and card.objectName() == "yearCard"):
            card = card.parent()

        if card and card.objectName() == "yearCard":
            self.periods_layout.removeWidget(card)
            card.deleteLater()
        else:
            # fallback
            self.periods_layout.removeWidget(period_row)
            period_row.deleteLater()

        self._sync_year_checkbox_with_rows(target_year=year_being_removed)
        self._update_load_button_state()

    def _get_years_with_rows(self) -> set[int]:

    def _load_selected_periods(self):
        """Recursive search for 'Cost Control ZAG 02_2026_working' + SAP file"""
        source_folder = self.source_folder
        dest_folder = self.dest_folder
        print(f"[DEBUG] Source folder (SAP) = {source_folder}")
        print(f"[DEBUG] Destination folder (Cost Control) = {dest_folder}")
        if not source_folder.is_dir():
            QMessageBox.warning(self, "Invalid Folder", "Source folder is not set or does not exist.")
            return
        if not dest_folder.is_dir():
            QMessageBox.warning(self, "Invalid Folder", "Destination folder is not set or does not exist.")
            return

        selected_periods = []
        for row in self._get_all_period_rows():
            year = row.year
            for month_name, cb in row.month_checkboxes.items():
                if cb.isChecked():
                    selected_periods.append((month_name, year))

        if not selected_periods:
            QMessageBox.warning(self, "No Selection", "Please select at least one month.")
            return

        new_periods = [p for p in selected_periods if p not in self.loaded_periods]
        if not new_periods:
            Toast(self, "All selected periods are already loaded")
            return

        # Remember selected periods to limit mapping population
        self._last_selected_periods = list(selected_periods)

        self.btn_load.setEnabled(False)

        total_steps = max(len(new_periods), 1)
        self._progress_dialog = QProgressDialog("Loading selected periods...", None, 0, total_steps, self)
        self._progress_dialog.setWindowTitle("Loading")
        self._progress_dialog.setWindowModality(Qt.WindowModality.NonModal)
        self._progress_dialog.setMinimumWidth(420)
        self._progress_dialog.setAutoClose(True)
        self._progress_dialog.setMinimumDuration(0)
        self._progress_dialog.setStyleSheet(
            "QProgressBar {border: 1px solid #D1D5DB; border-radius: 4px; text-align: center; height: 14px;}"
            "QProgressBar::chunk {background-color: #22C55E; border-radius: 4px;}"
        )
        self._progress_dialog.setValue(0)
        self._progress_dialog.show()
        self._progress_dialog.raise_()
        self._progress_dialog.activateWindow()

        self._load_thread = QThread()
        self._load_worker = LoadWorker(
            new_periods,
            str(source_folder),
            str(dest_folder)
        )

        self._load_worker.moveToThread(self._load_thread)
        self._load_thread.started.connect(self._load_worker.run)
        self._load_worker.progress.connect(self._on_load_progress)
        self._load_worker.result.connect(self._on_load_result)
        self._load_worker.error.connect(self._on_load_error)
        self._load_worker.finished.connect(self._load_thread.quit)
        self._load_worker.finished.connect(self._load_worker.deleteLater)
        self._load_thread.finished.connect(self._load_thread.deleteLater)

        self._load_thread.start()

    def _on_load_progress(self, step: int, message: str):
        self._progress_dialog.setValue(step)
        percent = int((step / max(1, self._progress_dialog.maximum())) * 100)
        self._progress_dialog.setLabelText(f"{message} ({percent}%)")

    def _on_load_result(self, result: dict):
        loaded = result.get('loaded', {})
        missing = result.get('missing', [])
        errors = result.get('errors', [])

        self.loaded_workbooks.update(loaded)
        if loaded:
            self.loaded_periods.update((month, year) for month, year, _ in loaded.keys())

        # Convert loaded dict to list format for the table
        loaded_list = [(month, year, info.get('path', '').split('\\')[-1] if isinstance(info, dict) else str(info))
                       for (month, year, ftype), info in loaded.items()]

        self._update_files_table(loaded_list, missing, errors)

        self._progress_dialog.setValue(self._progress_dialog.maximum())
        self._progress_dialog.setLabelText("Load complete")
        self.period_status.setText("Load complete")
        self.btn_load.setEnabled(True)
        QTimer.singleShot(300, self._populate_mappings_after_load)
        QTimer.singleShot(400, lambda: self._show_load_summary_dialog(loaded_list, missing, errors))

    def _on_load_error(self, msg: str):
        self._progress_dialog.close()
        self.btn_load.setEnabled(True)
        QMessageBox.critical(self, "Load Error", f"Error during loading:\n{msg}")

    def _populate_mappings_after_load(self):
        """Creates 4 cards per month/year with correct file types"""
        self._clear_all_mapping_rows()

        if not self.loaded_workbooks:
            self._show_waiting_state()
            return

        # Get unique (month_name, year) pairs ONLY from successfully loaded workbooks
        unique_periods = set()
        for key in self.loaded_workbooks.keys():
            month_name, year, _ = key
            unique_periods.add((month_name, year))

        created = 0
        try:
            for month_name, year in sorted(unique_periods):
                print(f"\n[DEBUG] === Processing {month_name} {year} ===")

                # Load month-specific mappings
                sources = get_mappings_for_month_year(month_name, year)

                # === 1-3: Dynamic cards (Cost Control   Consolidated) ===
                for src in sources:
                    src_dest_col = src.get("dest_column", "G")
                    print(f"[DEBUG] Dynamic card: {src['source_sheet']} {src_dest_col}")
                    self._add_mapping_row(
                        source_sheet=src["source_sheet"],
                        source_column=src["source_column"],
                        source_rows=src.get("source_rows", []),
                        dest_sheet=src["dest_sheet"],
                        dest_column=src_dest_col,
                        dest_rows=src.get("dest_rows", []),
                        month_name=month_name,
                        year=year,
                        mappings_file=MAPPINGS_BUDGET_REFI_PREV_YEAR,
                        source_file_type="cost_control",
                        save_json=False
                    )
                    created += 1

                # === 4: Classic SAP card from mappings.json (use mappings.json only for 2026+) ===
                classic_file = MAPPINGS_FILE if year >= 2026 else MAPPINGS_OLD_FILE
                classic_mappings = self._load_mappings_from_file(classic_file)
                matched = next((e for e in classic_mappings
                                if str(e.get("month", "")).lower().strip() == month_name.lower().strip()), None)

                classic_dest_col = str(matched.get("dest_column", "G")) if matched else "G"
                print(f"[DEBUG] Classic SAP card {classic_dest_col}")
                print(f"[DEBUG] Using file: {classic_file}, month: {month_name}, year: {year}")
                print(f"[DEBUG] matched: {matched}")
                self._add_mapping_row(
                    source_sheet=str(matched.get("source_sheet", "Report")) if matched else (
                        "Report" if year > 2025 else str(MONTH_TO_NUM.get(month_name, "1")).lstrip("0") or "1"),
                    source_column=str(matched.get("source_column", "N")) if matched else "N",
                    source_rows=matched.get("source_rows", []) if matched else [],
                    dest_sheet=str(matched.get("dest_sheet", "MZLZ Consolidated")) if matched else "MZLZ Consolidated",
                    dest_column=classic_dest_col,
                    dest_rows=matched.get("dest_rows", []) if matched else [],
                    month_name=month_name,
                    year=year,
                    mappings_file=classic_file,
                    source_file_type="sap",
                    save_json=False
                )
                created += 1

                # === 5: PAX/Traffic card from pax.json ===
                pax_file = MAPPINGS_PAX_FILE
                pax_mappings = self._load_mappings_from_file(pax_file)
                pax_matched = next(
                    (e for e in pax_mappings if str(e.get("month", "")).lower().strip() == month_name.lower().strip()),
                    None
                )
                pax_source_sheet = ""
                pax_source_column = "D"
                pax_source_rows = []
                pax_dest_sheet = "MZLZ Consolidated"
                pax_dest_column = "G"
                pax_dest_rows = []
                if pax_matched:
                    pax_source_sheet = str(pax_matched.get("source_sheet", "")).replace("{year}", str(year))
                    pax_source_column = str(pax_matched.get("source_column", "D"))
                    pax_source_rows = pax_matched.get("source_rows", [])
                    pax_dest_sheet = str(pax_matched.get("dest_sheet", "MZLZ Consolidated"))
                    pax_dest_column = str(pax_matched.get("dest_column", "G"))
                    pax_dest_rows = pax_matched.get("dest_rows", [])

                self._add_mapping_row(
                    source_sheet=pax_source_sheet,
                    source_column=pax_source_column,
                    source_rows=pax_source_rows,
                    dest_sheet=pax_dest_sheet,
                    dest_column=pax_dest_column,
                    dest_rows=pax_dest_rows,
                    month_name=month_name,
                    year=year,
                    mappings_file=pax_file,
                    source_file_type="pax",
                    save_json=False
                )
                created += 1

                # === 6: STAFF/FTE card from staff.json ===
                created += self._add_staff_mapping_rows(month_name, year)

                # === 7: SAP YTD Report card (JG column) ===
                # Source: YTD_{MM}_{YYYY}.xlsx   sheet "Report"   column C
                # Dest:   MZLZ Consolidated   column JG (always same column)
                sap_ytd_file = "mappings_sap_ytd.json"
                sap_ytd_mappings = self._load_mappings_from_file(Path(sap_ytd_file))
                sap_ytd_match = next(
                    (m for m in sap_ytd_mappings
                     if m.get('month') == month_name),
                    None
                )
                if sap_ytd_match:
                    sap_ytd_match = dict(sap_ytd_match)
                    sap_ytd_match['month'] = month_name
                    sap_ytd_match['year'] = year
                    sap_ytd_match['source_file_type'] = 'sap_ytd'
                    self._add_mapping_row(
                        source_sheet="Report",
                        source_column="C",
                        source_rows=sap_ytd_match.get('source_rows', []),
                        dest_sheet="MZLZ Consolidated",
                        dest_column="JG",
                        dest_rows=sap_ytd_match.get('dest_rows', []),
                        month_name=month_name,
                        year=year,
                        mappings_file=Path(sap_ytd_file),
                        source_file_type="sap_ytd",
                        save_json=False
                    )
                    created += 1
                    print(f"[DEBUG] SAP YTD card: {month_name} {year}JG")

                # === 8: YTD Cumulative card (IP-IZ columns) ===
                # IP=Jan, IQ=Jan+Feb, IR=+Mar, IS=+Apr, IT=+May, IU=+Jun,
                # IV=+Jul, IW=+Aug, IX=+Sep, IY=+Oct, IZ=+Nov. December has no YTD column.
                MONTH_TO_YTD_COL = {
                    'January': ('IP', None),  # IP = G (no prev YTD for Jan)
                    'February': ('IQ', 'IP'),
                    'March': ('IR', 'IQ'),
                    'April': ('IS', 'IR'),
                    'May': ('IT', 'IS'),
                    'June': ('IU', 'IT'),
                    'July': ('IV', 'IU'),
                    'August': ('IW', 'IV'),
                    'September': ('IX', 'IW'),
                    'October': ('IY', 'IX'),
                    'November': ('IZ', 'IY'),
                    'December': (None, None),  # No YTD column for December
                }
                ytd_col, prev_ytd_col = MONTH_TO_YTD_COL.get(month_name, (None, None))
                if ytd_col and matched:
                    # Include SAP dest_rows + PAX rows + Staff row in YTD sum
                    sap_dest_rows = matched.get("dest_rows", []) if matched else []
                    pax_dest_rows = [5, 7, 8, 9, 10, 11]
                    staff_dest_rows = [16]
                    all_ytd_rows = sorted(set(sap_dest_rows + pax_dest_rows + staff_dest_rows))
                    self._add_mapping_row(
                        source_sheet=str(matched.get("source_sheet", "")) if matched else "",
                        source_column=str(matched.get("source_column", "D")) if matched else "D",
                        source_rows=matched.get("source_rows", []) if matched else [],
                        dest_sheet=str(
                            matched.get("dest_sheet", "MZLZ Consolidated")) if matched else "MZLZ Consolidated",
                        dest_column=ytd_col,
                        dest_rows=all_ytd_rows,
                        month_name=month_name,
                        year=year,
                        mappings_file=classic_file,
                        source_file_type="ytd",
                        save_json=False
                    )
                    created += 1
                    print(
                        f"[DEBUG] YTD card: {month_name}{ytd_col} (prev: {prev_ytd_col}) | {len(all_ytd_rows)} rows incl. PAX+Staff")

            print(f"[DEBUG] TOTAL cards created: {created}")

            if created > 0:
                if hasattr(self, "_progress_dialog") and self._progress_dialog is not None:
                    self._progress_dialog.close()
                Toast(self, f"{created} cards created (month-specific columns)")
                QTimer.singleShot(450, self._show_mappings_selection_popup)
            else:
                self._show_waiting_state()

        except Exception as e:
            traceback.print_exc()
            QMessageBox.critical(self, "Error", f"Failed to create mappings:\n{str(e)}")

    def _undock_mappings(self):
        """Open all current mappings in a separate popup window"""
        if not self.mapping_rows:
            QMessageBox.information(self, "Nothing to undock", "No mappings loaded yet.")
            return

        # Prevent multiple popups
        if hasattr(self, '_popup') and self._popup is not None and self._popup.isVisible():
            self._popup.raise_()
            self._popup.activateWindow()
            return

        self._popup = MappingPopup(self, self.mapping_rows[:])  # copy list
        self._popup.show()
        if hasattr(self, "_progress_dialog") and self._progress_dialog is not None:
            self._progress_dialog.close()
        Toast(self, f"{len(self.mapping_rows)} mappings undocked to popup")

    def _add_mapping_row(self, source_sheet="Report", source_column="D", source_rows=None,
                         dest_sheet="MZLZ Consolidated", dest_column="G", dest_rows=None,
                         month_name=None, year=None, mappings_file=None,
                         source_file_type="cost_control",  #   ADD THIS
                         save_json=True):

        if source_rows is None: source_rows = []
        if dest_rows is None:   dest_rows = []

        # Auto-select file if none passed
        if mappings_file is None:
            mappings_file = MAPPINGS_OLD_FILE if (year is not None and year <= 2025) else MAPPINGS_FILE

        self.mapping_counter += 1

        row = MappingRow(
            index=self.mapping_counter,
            on_remove=self._remove_mapping_with_password,
            on_run=self.execute_single,
            on_change=self._save_mappings_to_json if save_json else None,
            source_sheet=source_sheet,
            source_column=source_column,
            source_rows=source_rows,
            dest_sheet=dest_sheet,
            dest_column=dest_column,
            dest_rows=dest_rows,
            month_name=month_name,
            year=year,
            mappings_file=mappings_file,
            source_file_type=source_file_type,
            mapping_source_label=mappings_file.name if mappings_file else ""
        )

        self.mapping_rows.append(row)
        self._add_mapping_row_to_group(row)

        # Safely set label
        if hasattr(row, 'mappings_label') and mappings_file:
            row.mappings_label.setText(f"Using: {mappings_file.name}")

        if save_json:
            self._save_mappings_to_json()

        return row

    def _transfer(self, mappings, deferred_save=False):
        """3 cards: read+write in Cost Control file
           4th card: read from SAP, write to Cost Control file

           If copy_full_sheet is True: do cell transfer AND sheet copy
           If copy_full_sheet is False: only do cell transfer"""

        print(f"[DEBUG] _transfer called, copy_full_sheet = {mappings.get('copy_full_sheet', False)}")
        print(
            f"[DEBUG] source_sheet = '{mappings.get('source_sheet', '')}', custom_sheet_name = '{mappings.get('custom_sheet_name', '')}'")

        do_sheet_copy = mappings.get('copy_full_sheet', False)

        print(f"[DEBUG] Final do_sheet_copy = {do_sheet_copy}")

        month_year = (mappings['month'], mappings['year'])
        read_type = mappings.get('source_file_type', 'cost_control')
        write_month = mappings.get('write_month', month_year[0])
        write_year = mappings.get('write_year', month_year[1])

        # === SAP YTD Report Transfer ===
        # Reads YTD_{MM}_{YYYY}.xlsx   sheet "Report"   column C   writes to JG in MZLZ Consolidated
        # Values are divided by -1000 same as SAP monthly
        if read_type == "sap_ytd":
            month_name = mappings['month']
            year = mappings['year']
            ytd_key = (month_name, year, 'sap_ytd')
            ytd_wb_entry = self.loaded_workbooks.get(ytd_key, {})
            ytd_wb = ytd_wb_entry.get('wb_values') or ytd_wb_entry.get('workbook')

            if not ytd_wb:
                print(f"[DEBUG SAP_YTD] No YTD workbook loaded for {month_name} {year}")
                return 0

            # Find "Report" sheet (case-insensitive)
            report_sheet = next(
                (s for s in ytd_wb.sheetnames if s.lower() == 'report'),
                ytd_wb.sheetnames[0] if ytd_wb.sheetnames else None
            )
            if not report_sheet:
                print(f"[DEBUG SAP_YTD] No 'Report' sheet found in YTD workbook")
                return 0

            ws_ytd = ytd_wb[report_sheet]

            # Get dest workbook
            write_key = (write_month, write_year, 'cost_control')
            if write_key not in self.loaded_workbooks:
                print(f"[DEBUG SAP_YTD] Dest workbook not loaded for {write_key}")
                return 0
            write_wb = self.loaded_workbooks[write_key]['workbook']
            mzlz_sheet = next(
                (s for s in write_wb.sheetnames if 'mzlz' in s.lower() and 'consol' in s.lower()),
                None
            )
            if not mzlz_sheet:
                print(f"[DEBUG SAP_YTD] MZLZ Consolidated sheet  ")
                return 0
            ws_dest = write_wb[mzlz_sheet]


            dest_column = mappings.get('dest_column', 'JG')
            source_column = mappings.get('source_column', 'C')
            cells_written = 0

            # New JSON format: rowMap = {dest_row: src_row}
            row_map = mappings.get('rowMap', {})
            print(f"[DEBUG SAP_YTD] rowMap entries: {len(row_map)} (month={month_name} {year})")
            # Show a few mappings to verify dest->source direction
            preview = list(row_map.items())[:5]
            if preview:
                print("[DEBUG SAP_YTD] rowMap preview (dest->source):", preview)
            for dst_row_str, src_row in row_map.items():
                try:
                    dst_row = int(dst_row_str)
                except (ValueError, TypeError):
                    continue

                if src_row == 0:
                    value = 0
                elif isinstance(src_row, list):
                    total = 0
                    for r in src_row:
                        cell_val = ws_ytd[f"{source_column}{r}"].value
                        if isinstance(cell_val, (int, float)):
                            total += cell_val
                    value = total / (-1000)
                else:
                    value = ws_ytd[f"{source_column}{src_row}"].value
                    if value is None:
                        value = 0
                    elif isinstance(value, (int, float)):
                        value = value / (-1000)

                ws_dest[f"{dest_column}{dst_row}"].value = value
                cells_written += 1

            # Extra JG rows sourced from other sheets in the destination workbook
            extra_jg_sources = [
                (5, "TRAFFIC mott 2025", "Q19", 1000),
                (7, "TRAFFIC mott 2025", "Q34", 1000),
                (8, "TRAFFIC mott 2025", "Q81", None),
                (9, "TRAFFIC mott 2025", "Q87", None),
                (10, "TRAFFIC mott 2025", "Q160", None),
                (11, "TRAFFIC mott 2025", "Q134", None),
                (16, "Staff report", "AC34", None),
            ]

            def _get_sheet_by_name(wb, name):
                for sheet_name in wb.sheetnames:
                    if sheet_name.lower() == name.lower():
                        return wb[sheet_name]
                return None

            wb_values_dest = self.loaded_workbooks.get(write_key, {}).get('wb_values')

            for dst_row, sheet_name, cell_ref, divisor in extra_jg_sources:
                src_sheet = _get_sheet_by_name(write_wb, sheet_name)
                src_sheet_values = _get_sheet_by_name(wb_values_dest, sheet_name) if wb_values_dest else None
                value = 0
                if src_sheet is None and src_sheet_values is None:
                    print(f"[DEBUG SAP_YTD] Sheet '{sheet_name}' not found for JG row {dst_row}")
                else:
                    raw = None
                    if src_sheet_values is not None:
                        raw = src_sheet_values[cell_ref].value
                    if raw is None and src_sheet is not None:
                        raw = src_sheet[cell_ref].value
                    if raw is None:
                        value = 0
                    elif isinstance(raw, (int, float)):
                        value = raw
                    else:
                        try:
                            value = float(str(raw).replace(',', '.'))
                        except Exception:
                            value = 0
                    if divisor and isinstance(value, (int, float)):
                        value = value / divisor
                ws_dest[f"{dest_column}{dst_row}"].value = value
                cells_written += 1

            print(f"[DEBUG SAP_YTD] {month_name} {year}: direct copy -> {dest_column} | {cells_written} rows written")

            if not deferred_save:
                write_path = self.loaded_workbooks[write_key]['path']
                write_wb.save(write_path)

            return cells_written

        # === YTD Transfer ===
        # YTD column = current month dest column + previous YTD column (both already in dest file)
        if read_type == "ytd":
            # Force YTD writes into JG via sap_ytd rowMap
            try:
                sap_ytd_mappings = self._load_mappings_from_file(Path("mappings_sap_ytd.json"))

                target_month = mappings.get('write_month') or mappings.get('month')
                target_year = mappings.get('write_year') or mappings.get('year')

                # Normalize numeric month (e.g., "07" or 7) to month name
                if isinstance(target_month, (int, float)):
                    target_month = str(int(target_month)).zfill(2)
                if isinstance(target_month, str) and target_month.isdigit():
                    num_to_name = {v: k for k, v in MONTH_TO_NUM.items()} if 'MONTH_TO_NUM' in globals() else {}
                    target_month = num_to_name.get(target_month.zfill(2), target_month)

                sap_match = next(
                    (
                        m for m in sap_ytd_mappings
                        if m.get('month') == target_month and (
                            not m.get('year') or str(m.get('year', '')) == str(target_year)
                        )
                    ),
                    None
                )

                if sap_match and sap_match.get('rowMap'):
                    redirected = dict(mappings)
                    redirected.update({
                        'source_file_type': 'sap_ytd',
                        'source_sheet': sap_match.get('source_sheet', 'Report'),
                        'source_column': sap_match.get('source_column', 'C'),
                        'dest_sheet': sap_match.get('dest_sheet', 'MZLZ Consolidated'),
                        'dest_column': sap_match.get('dest_column', 'JG'),
                        'rowMap': sap_match.get('rowMap', {}),
                    })
                    print(
                        f"[DEBUG] Redirecting YTD to sap_ytd for {target_month} {target_year} -> {redirected.get('dest_column')}"
                    )
                    return self._transfer(redirected, deferred_save=deferred_save)
            except Exception as e:
                print(f"[DEBUG] YTD->sap_ytd redirect failed: {e}")

            MONTH_TO_YTD_COL = {
                'January': ('IP', None),
                'February': ('IQ', 'IP'),
                'March': ('IR', 'IQ'),
                'April': ('IS', 'IR'),
                'May': ('IT', 'IS'),
                'June': ('IU', 'IT'),
                'July': ('IV', 'IU'),
                'August': ('IW', 'IV'),
                'September': ('IX', 'IW'),
                'October': ('IY', 'IX'),
                'November': ('IZ', 'IY'),
                'December': (None, None),
            }
            MONTH_TO_DEST_COL = {
                'January': 'G', 'February': 'W', 'March': 'AM',
                'April': 'BD', 'May': 'BW', 'June': 'CP',
                'July': 'DJ', 'August': 'EF', 'September': 'FB',
                'October': 'FY', 'November': 'GX', 'December': 'HW',
            }
            month_name = mappings['month']
            ytd_col, prev_ytd_col = MONTH_TO_YTD_COL.get(month_name, (None, None))
            if not ytd_col:
                return 0

            curr_col = MONTH_TO_DEST_COL.get(month_name, 'G')

            # Get destination workbook
            write_key = (write_month, write_year, 'cost_control')
            if write_key not in self.loaded_workbooks:
                print(f"[DEBUG YTD] Dest workbook not loaded for {write_key}")
                return 0
            write_wb = self.loaded_workbooks[write_key]['workbook']

            # Find MZLZ Consolidated sheet (case-insensitive fallback)
            mzlz_sheet = next(
                (s for s in write_wb.sheetnames if 'mzlz' in s.lower() and 'consol' in s.lower()),
                None
            )
            if not mzlz_sheet:
                print(f"[DEBUG YTD] 'MZLZ Consolidated' sheet   in workbook. Sheets: {write_wb.sheetnames}")
                return 0
            ws_dest = write_wb[mzlz_sheet]
            print(f"[DEBUG YTD] Writing to sheet: '{mzlz_sheet}'")

            dest_rows = mappings.get('dest_rows', [])
            cells_written = 0

            # Recalculate current column summary rows before YTD
            self._calculate_mzlz_summary_rows(ws_dest, curr_col)

            SUMMARY_ROWS = {
                6, 13, 33, 54, 66, 68, 69, 70, 71, 76, 78,
                82, 83, 90, 94, 96, 103, 105, 107, 130, 136,
                150, 152, 154, 156, 157, 158, 170, 172, 174,
                175, 181, 214, 218, 222, 226
            }

            for dest_row in dest_rows:
                if dest_row in SUMMARY_ROWS:
                    continue
                # Current month value already written in dest
                try:
                    curr_val = float(ws_dest[f"{curr_col}{dest_row}"].value or 0)
                except (ValueError, TypeError):
                    curr_val = 0.0

                # Previous YTD value (0 for January)
                prev_val = 0.0
                if prev_ytd_col:
                    try:
                        prev_val = float(ws_dest[f"{prev_ytd_col}{dest_row}"].value or 0)
                    except (ValueError, TypeError):
                        prev_val = 0.0

                ytd_val = curr_val + prev_val
                if dest_row in [23, 24, 25, 26, 33, 54]:
                    print(f"[DEBUG YTD SUM] row={dest_row}: {curr_col}={curr_val} + {prev_ytd_col or 'nothing'}={prev_val} = {ytd_val}{ytd_col}")
                ws_dest[f"{ytd_col}{dest_row}"].value = ytd_val
                cells_written += 1

            # Recalculate summary rows from YTD column itself
            self._calculate_mzlz_summary_rows(ws_dest, ytd_col)

            print(
                f"[DEBUG YTD] {month_name}: {curr_col} + {prev_ytd_col or 'nothing'}{ytd_col} | {cells_written} rows written")

            if not deferred_save:
                write_path = self.loaded_workbooks[write_key]['path']
                write_wb.save(write_path)

            return cells_written

        # Read source
        read_key = (month_year[0], month_year[1], read_type)
        if read_key not in self.loaded_workbooks:
            # Fallback for legacy keys without file type
            legacy_key = (month_year[0], month_year[1])
            if legacy_key in self.loaded_workbooks and read_type == "cost_control":
                read_wb = self.loaded_workbooks[legacy_key]['workbook']
            else:
                print(f"[WARN] Read file not loaded: {read_type} (skipping transfer)")
                return 0
        else:
            read_wb = self.loaded_workbooks[read_key]['workbook']

        # Use pre-cached data_only workbook (loaded in parallel at file-load time)
        # Falls back to a fresh load only if the cache entry is missing for some reason
        read_wb_values = self.loaded_workbooks[read_key].get('wb_values')
        if read_wb_values is None:
            source_path = self.loaded_workbooks[read_key]['path']
            try:
                read_wb_values = openpyxl.load_workbook(source_path, data_only=True, keep_vba=True)
                # Store it so subsequent transfers don't reload either
                self.loaded_workbooks[read_key]['wb_values'] = read_wb_values
                print(f"[DEBUG] wb_values cache miss    loaded fresh for {source_path}")
            except Exception as e:
                print(f"[DEBUG] Could not load data_only workbook: {e}")
                read_wb_values = None

        # Always write to Cost Control file (allow override for batch export)
        write_key = (write_month, write_year, "cost_control")
        if write_key not in self.loaded_workbooks:
            raise ValueError("Cost Control file not loaded")

        write_wb = self.loaded_workbooks[write_key]['workbook']
        write_path = self.loaded_workbooks[write_key]['path']

        sheet_name = mappings['source_sheet'].strip()

        print(f"[DEBUG] Read from {read_type} | Sheet '{sheet_name}' | Write to Cost Control")

        resolved_sheet = sheet_name

        def normalize_sheet(name: str) -> str:
            return re.sub(r"[^a-z0-9]", "", name.lower())

        if resolved_sheet not in read_wb.sheetnames:
            # Try common fallbacks (trim "Worksheet ", case/space-insensitive, prev-year toggle)
            candidates = {normalize_sheet(s.strip()): s for s in read_wb.sheetnames}
            trimmed = sheet_name.replace("Worksheet ", "").strip()
            normalized_trimmed = normalize_sheet(trimmed)

            if normalized_trimmed in candidates:
                resolved_sheet = candidates[normalized_trimmed]
            else:
                # Case/space-insensitive match
                match = next((s for s in read_wb.sheetnames if normalize_sheet(s) == normalized_trimmed), None)
                if match:
                    resolved_sheet = match
                elif read_type == "sap" and len(read_wb.sheetnames) == 1:
                    # SAP file often has a single sheet; use it
                    resolved_sheet = read_wb.sheetnames[0]
                else:
                    # Try year fallback (e.g., 2025 -> 2026 or 2026 -> 2025)
                    year_str = str(mappings.get('year', ''))
                    if year_str and year_str in trimmed:
                        alt_year = str(int(year_str) - 1)
                        alt = trimmed.replace(year_str, alt_year)
                        match_alt = next(
                            (s for s in read_wb.sheetnames if normalize_sheet(s) == normalize_sheet(alt)),
                            None
                        )
                        if match_alt:
                            resolved_sheet = match_alt

                    # Budget fallback: "2026 Budget" -> "BUDGET 2026"
                    if "budget" in trimmed.lower():
                        budget_alt = f"BUDGET {year_str}".strip()
                        match_budget = next(
                            (s for s in read_wb.sheetnames if normalize_sheet(s) == normalize_sheet(budget_alt)),
                            None
                        )
                        if match_budget:
                            resolved_sheet = match_budget

        if resolved_sheet not in read_wb.sheetnames:
            print(f"[DEBUG] Sheet '{sheet_name}'   in {read_type} workbook. Skipping.")
            print(f"[DEBUG] Available sheets: {', '.join(read_wb.sheetnames)}")
            return 0

        # Get values version of the sheet
        ws_source = read_wb[resolved_sheet]
        ws_source_values = None
        if read_wb_values and resolved_sheet in read_wb_values.sheetnames:
            ws_source_values = read_wb_values[resolved_sheet]

        ws_dest = write_wb["MZLZ Consolidated"]
        dest_column = mappings.get('dest_column', '').upper()
        source_column = mappings.get('source_column', 'A')
        source_sheet_name = mappings.get('source_sheet', '').strip()
        # Always use the actually-resolved sheet name so formula evaluation
        # finds the correct sheet even if the file was renamed (e.g. "02_2025" instead of "1").
        sheet_ref_name = resolved_sheet

        xlwings_app = None
        xlwings_book = None
        xlwings_sheet = None
        xlwings_failed = False
        xlwings_calculated = False
        xlwings_owned_app = False
        xlwings_owned_book = False
        xlwings_path = None
        if read_type == "cost_control":
            xlwings_path = self.loaded_workbooks.get(read_key, {}).get('path')

        def _get_xlwings_value(cell_ref: str):
            nonlocal xlwings_app, xlwings_book, xlwings_sheet, xlwings_failed, xlwings_calculated
            nonlocal xlwings_owned_app, xlwings_owned_book
            if xlwings_failed or not xlwings_path:
                return None
            if xlwings_book is None:
                try:
                    file_name = Path(xlwings_path).name
                    # Try to attach to already-open workbook
                    for app in xw.apps:
                        for book in app.books:
                            if book.name.lower() == file_name.lower():
                                xlwings_app = app
                                xlwings_book = book
                                xlwings_sheet = xlwings_book.sheets[resolved_sheet]
                                print(f"[DEBUG] xlwings: attached to already-open '{file_name}'")
                                break
                        if xlwings_book is not None:
                            break

                    # If not already open, open in existing app or create one
                    if xlwings_book is None:
                        if xw.apps:
                            xlwings_app = xw.apps[0]
                        else:
                            xlwings_app = xw.App(visible=False, add_book=False)
                            xlwings_app.display_alerts = False
                            xlwings_app.screen_updating = False
                            xlwings_owned_app = True
                        xlwings_book = xlwings_app.books.open(xlwings_path, update_links=False)
                        xlwings_owned_book = True
                        xlwings_sheet = xlwings_book.sheets[resolved_sheet]
                        print(f"[DEBUG] xlwings: opened '{file_name}'")
                except Exception as e:
                    xlwings_failed = True
                    print(f"[DEBUG] xlwings fallback failed: {e}")
                    return None
            if not xlwings_calculated:
                try:
                    xlwings_book.app.calculate()
                except Exception as e:
                    print(f"[DEBUG] xlwings calculate failed: {e}")
                xlwings_calculated = True
            try:
                return xlwings_sheet.range(cell_ref).value
            except Exception as e:
                print(f"[DEBUG] xlwings read failed for {cell_ref}: {e}")
                return None

        cells_written = 0
        for index, (src_row, dest_row) in enumerate(zip(mappings['source_rows'], mappings['dest_rows'])):
            expected_name = None
            if mappings.get('names') and index < len(mappings['names']):
                expected_name = str(mappings['names'][index]).strip()

            resolved_src_row = src_row
            missing_source = False
            if expected_name:
                resolved_src_row = self._resolve_row_by_name(
                    ws_source,
                    mappings['source_column'],
                    src_row,
                    expected_name,
                    mappings.setdefault('unresolved_names', [])
                )
                if resolved_src_row is None:
                    missing_source = True

            # Validate destination label in column B (case/space-insensitive)
            # Skip validation for PAX (different structure) and cost_control
            dest_expected = self.destination_label_map.get(int(dest_row)) if mappings.get('source_file_type') not in (
                'pax', 'cost_control') else None
            if dest_expected:
                resolved_dest_row = self._resolve_dest_row_by_label(
                    ws_dest,
                    dest_row,
                    dest_expected,
                    self.missing_destination_labels
                )
                if resolved_dest_row is None:
                    continue
            else:
                resolved_dest_row = dest_row

            # Debug: Show source info for SAP
            if mappings.get('source_file_type') == 'sap':
                print(
                    f"[DEBUG SAP] source_sheet={mappings['source_sheet']}, source_column={mappings['source_column']}, source_row={resolved_src_row}, dest_column={mappings['dest_column']}, dest_row={resolved_dest_row}")
                print(
                    f"[DEBUG SAP] source_rows count={len(mappings.get('source_rows', []))}, dest_rows count={len(mappings.get('dest_rows', []))}")

            # Debug output for PAX
            if mappings.get('source_file_type') == 'pax':
                print(
                    f"[DEBUG PAX] index={index}, src_row={resolved_src_row}, dest_row={resolved_dest_row}, source_rows={mappings.get('source_rows')}, dest_rows={mappings.get('dest_rows')}")

            # Optimized: Direct cell access with faster value retrieval
            src_cell = ws_source[f"{mappings['source_column']}{resolved_src_row}"]

            # Get value - ALWAYS use data_only workbook to avoid copying formulas
            value = None
            if ws_source_values:
                try:
                    values_cell = ws_source_values[f"{mappings['source_column']}{resolved_src_row}"]
                    value = values_cell.value
                    if mappings.get('source_file_type') == 'sap':
                        print(f"[DEBUG SAP VALUE] row={resolved_src_row}, data_only value={value}")
                    if value in (0, 1) and resolved_src_row < 150:
                        print(
                            f"[DEBUG CACHE CHECK] row={resolved_src_row}, col={mappings['source_column']}, cached_value={value}, "
                            f"sheet_values='{ws_source_values.title}', sheet_source='{ws_source.title}'")
                except Exception as e:
                    print(f"[DEBUG SAP VALUE ERROR] {e}")
                    pass

            # If cache is missing OR suspicious, fall back to Excel-calculated value (xlwings) for formulas
            if not missing_source:
                if src_cell.data_type == 'f' and mappings.get('source_file_type') == 'cost_control':
                    if value is None or (isinstance(value, (int, float)) and value in (0, 1)):
                        xlwings_value = _get_xlwings_value(f"{mappings['source_column']}{resolved_src_row}")
                        if xlwings_value is not None:
                            print(
                                f"[DEBUG XLWINGS OVERRIDE] row={resolved_src_row}, cached={value}, xlwings={xlwings_value}, sheet={resolved_sheet}")
                            value = xlwings_value
                    elif resolved_sheet and 'budget' in resolved_sheet.lower():
                        value = src_cell.value
                        if resolved_dest_row in {142,144,145,146}:
                            print(
                                f"[DEBUG BUDGET FORMULA COPY] sheet={resolved_sheet} cell={mappings['source_column']}{resolved_src_row} "
                                f"formula={src_cell.value}")
                elif value is None and src_cell.data_type != 'f':  # Only copy if NOT a formula
                    value = src_cell.value
                    if mappings.get('source_file_type') == 'sap':
                        print(
                            f"[DEBUG SAP VALUE FALLBACK] row={resolved_src_row}, value={value}, data_type={src_cell.data_type}")
                elif value is None and resolved_sheet and 'budget' in resolved_sheet.lower():
                    value = src_cell.value
                    if resolved_dest_row in {142,144,145,146}:
                        print(
                            f"[DEBUG BUDGET FORMULA COPY] sheet={resolved_sheet} cell={mappings['source_column']}{resolved_src_row} "
                            f"formula={src_cell.value}")

            # If no source match/value, explicitly write 0 to avoid blank cells
            if value is None:
                value = 0

            # Divide by 1000: SAP always, PAX only first 2 rows (indices 0,1 = dest_rows 5,7)
            source_type = mappings.get('source_file_type')
            if source_type == 'sap' and value is not None and isinstance(value, (int, float)):
                value = value / -1000
            elif source_type == 'pax' and index < 2 and value is not None and isinstance(value, (int, float)):
                # First 2 PAX rows (5, 7) get divided by 1000
                value = value / 1000

            # Optimized: Direct assignment (ALWAYS write, not just when value was None)
            if mappings.get('source_file_type') == 'cost_control' and resolved_dest_row in {142,144,145,146,162,167,168,187,188,192,200,201,203,204,207}:
                print(
                    f"[DEBUG CC ROW] sheet={resolved_sheet} col={mappings['source_column']} src_row={resolved_src_row} "
                    f"dest={mappings['dest_column']}{resolved_dest_row} value={value}"
                )
            print(f"[DEBUG] Writing to column: {mappings['dest_column']}, row: {resolved_dest_row}, value: {value}")
            dest_cell = ws_dest[f"{mappings['dest_column']}{resolved_dest_row}"]
            dest_cell.value = value
            cells_written += 1

        if xlwings_book is not None and xlwings_owned_book:
            try:
                xlwings_book.close()
            except Exception:
                pass
        if xlwings_app is not None and xlwings_owned_app:
            try:
                xlwings_app.quit()
            except Exception:
                pass

        # Post-pass: evaluate formulas in the destination column and replace with values
        # For cost_control source types, SUMIFS/VLOOKUP formula rows (118-123, 142-147) reference
        # the SAP sheet - so we need to pass the SAP worksheet, not the Cost Control sheet.
        if dest_column:
            if read_type == "cost_control":
                # Rows 118-123 and 142-147 use SUMIFS/VLOOKUP against the PREVIOUS MONTH's
                # SAP sheet    which is a sheet inside the current SAP monthly export file.
                # e.g. for June   sheet '5' (May) inside 06_2025.xlsx
                month_name = mappings.get('month', '')
                year = mappings.get('year', 0)
                month_num_map = {
                    'January': '1', 'February': '2', 'March': '3', 'April': '4',
                    'May': '5', 'June': '6', 'July': '7', 'August': '8',
                    'September': '9', 'October': '10', 'November': '11', 'December': '12'
                }
                prev_month_num_map = {
                    'January': None, 'February': '1', 'March': '2', 'April': '3',
                    'May': '4', 'June': '5', 'July': '6', 'August': '7',
                    'September': '8', 'October': '9', 'November': '10', 'December': '11'
                }
                m_num = month_num_map.get(month_name, '')
                m_num_padded = m_num.zfill(2) if m_num else ''
                prev_m_num = prev_month_num_map.get(month_name)
                prev_m_num_padded = prev_m_num.zfill(2) if prev_m_num else ''

                # Look up current SAP workbook (contains all month sheets)
                sap_key = (month_name, year, 'sap')
                sap_wb_entry = self.loaded_workbooks.get(sap_key, {})
                sap_wb = sap_wb_entry.get('wb_values') or sap_wb_entry.get('workbook')
                if sap_wb and prev_m_num:
                    # Find previous month's sheet inside current SAP file
                    prev_candidates = [
                        prev_m_num,
                        prev_m_num_padded,
                    ]
                    sap_sheet_name = next(
                        (s for s in prev_candidates if s in sap_wb.sheetnames),
                        None
                    )
                    if sap_sheet_name:
                        source_ws_for_formulas = sap_wb[sap_sheet_name]
                        sheet_ref_name = sap_sheet_name
                        print(
                            f"[DEBUG] _apply_fixed_formula_values using prev-month SAP sheet: '{sap_sheet_name}' for {month_name} {year}")
                    else:
                        # Fallback to current month sheet
                        curr_candidates = [m_num, m_num_padded, f"{m_num_padded}_{year}"]
                        sap_sheet_name = next(
                            (s for s in curr_candidates if s in sap_wb.sheetnames),
                            sap_wb.sheetnames[0]
                        )
                        source_ws_for_formulas = sap_wb[sap_sheet_name]
                        sheet_ref_name = sap_sheet_name
                        print(
                            f"[DEBUG] _apply_fixed_formula_values: prev-month sheet  , using '{sap_sheet_name}'")
                elif sap_wb and not prev_m_num:
                    # January    no previous month, use current month sheet
                    curr_candidates = [m_num, m_num_padded, f"{m_num_padded}_{year}"]
                    sap_sheet_name = next(
                        (s for s in curr_candidates if s in sap_wb.sheetnames),
                        sap_wb.sheetnames[0]
                    )
                    source_ws_for_formulas = sap_wb[sap_sheet_name]
                    sheet_ref_name = sap_sheet_name
                    print(f"[DEBUG] _apply_fixed_formula_values: January    using current sheet '{sap_sheet_name}'")
                else:
                    source_ws_for_formulas = ws_source_values or ws_source
                    print(f"[DEBUG] _apply_fixed_formula_values: no SAP found for {month_name} {year}")
            else:
                # For SAP source    ws_source is already the correct month sheet
                # Make sure sheet_ref_name matches ws_source.title (actual sheet name in file)
                source_ws_for_formulas = ws_source_values or ws_source
                if source_ws_for_formulas is not None:
                    try:
                        sheet_ref_name = source_ws_for_formulas.title
                        print(f"[DEBUG] _apply_fixed_formula_values using current SAP sheet: '{sheet_ref_name}'")
                    except Exception:
                        pass

            self._apply_fixed_formula_values(
                ws_dest,
                dest_column,
                sheet_ref_name,
                source_column,
                source_ws_for_formulas,
                allow_ytd_override=(mappings.get('source_file_type') == 'sap')
            )

        # Close the values workbook
        if read_wb_values:
            read_wb_values.close()

        # Optimized save - use faster write mode
        if not deferred_save:
            write_wb.calculation.calcMode = 'manual'
            print(f"[DEBUG] Saving workbook, type: {type(write_wb).__name__}")
            write_wb.save(write_path)
            write_wb.calculation.calcMode = 'auto'

            # Break external links AND restore cached values for all formula cells.
            # openpyxl strips <v> cached values on save    we re-inject them from wb_values
            # (the data_only snapshot of the destination file, loaded before any modifications).
            _write_wb_values = self.loaded_workbooks.get(write_key, {}).get('wb_values')
            break_external_links(write_path, wb_values=_write_wb_values)

            # Let real Excel recalculate everything so dependent formula cells
            # reflect the new values we just wrote (e.g. -1000 propagating through SUM/IF chains).
            # This also writes fresh <v> cached values for every formula cell.
            self._finalize_with_excel_calc(write_path)

            # Cleanup temp file if it exists
            write_info = self.loaded_workbooks.get(write_key, {})
            tmp = write_info.get('tmp_path')
            if tmp and Path(tmp).exists():
                try:
                    os.remove(tmp)
                    print(f"[DEBUG] Cleaned up temp file: {tmp}")
                except Exception:
                    pass

        # If copy_full_sheet is checked, ALSO copy the entire sheet (outside deferred_save block)
        if do_sheet_copy:
            try:
                sheet_cells = self._copy_full_sheet(mappings, deferred_save=deferred_save)
                cells_written += sheet_cells
            except Exception as e:
                print(f"[DEBUG] Sheet copy failed: {e}")

        return cells_written

    def _format_sheet_name_for_formula(self, name: str) -> str:
        if re.match(r"^[A-Za-z_][A-Za-z0-9_]*$", name):
            return name
        escaped = name.replace("'", "''")
        return f"'{escaped}'"

    def _replace_sheet_reference(self, formula: str, src_sheet: str, dest_sheet: str) -> str:
        if not isinstance(formula, str) or not formula.startswith("=") or src_sheet == dest_sheet:
            return formula
        src_escaped = re.escape(src_sheet)
        dest_formatted = self._format_sheet_name_for_formula(dest_sheet)

        def _replacement(_match):
            return f"{dest_formatted}!"

        updated = re.sub(rf"'{src_escaped}'!", _replacement, formula)
        updated = re.sub(rf"(<!')\b{src_escaped}!", _replacement, updated)
        if updated != formula:
            print(f"[DEBUG] Formula updated: {formula} -> {updated}")
        return updated

    def _repair_ref_formula(self, formula: str, sheet_name: str, source_column: Optional[str] = None) -> str:
        if not isinstance(formula, str) or "#REF!" not in formula:
            return formula

        base_col = (source_column or "A").upper()
        try:
            sum_col = get_column_letter(column_index_from_string(base_col) + 1)
        except Exception:
            sum_col = "B" if base_col == "A" else base_col

        sheet_ref = self._format_sheet_name_for_formula(sheet_name)
        criteria_range = f"{sheet_ref}!${base_col}:${base_col}"
        sum_range = f"{sheet_ref}!${sum_col}:${sum_col}"
        table_range = f"{sheet_ref}!${base_col}:${sum_col}"

        def replace_seq(replacements):
            idx = 0

            def _sub(_match):
                nonlocal idx
                rep = replacements[min(idx, len(replacements) - 1)]
                idx += 1
                return rep

            return re.sub(r"#REF!", _sub, formula)

        upper = formula.upper()
        if "SUMIFS(" in upper:
            updated = replace_seq([sum_range, criteria_range])
        elif "SUMIF(" in upper:
            updated = replace_seq([criteria_range, sum_range])
        elif "VLOOKUP(" in upper:
            updated = replace_seq([table_range])
        else:
            updated = replace_seq([criteria_range])

        if updated != formula:
            print(f"[DEBUG] #REF repair: {formula} -> {updated}")
        return updated

    def _split_formula_args(self, args_text: str) -> List[str]:
        args = []
        current = []
        depth = 0
        for ch in args_text:
            if ch == '(':
                depth += 1
            elif ch == ')':
                depth -= 1
            if depth == 0 and ch in (';', ','):
                arg = ''.join(current).strip()
                if arg:
                    args.append(arg)
                current = []
            else:
                current.append(ch)
        trailing = ''.join(current).strip()
        if trailing:
            args.append(trailing)
        return args

    def _parse_cell_reference(self, ref: str) -> Optional[Tuple[str, int]]:
        match = re.match(r"\$([A-Za-z]+)\$(\d+)", ref)
        if not match:
            return None
        return match.group(1).upper(), int(match.group(2))

    def _get_cell_value_from_ws(self, ws, ref: str) -> Optional[Any]:
        parsed = self._parse_cell_reference(ref)
        if not parsed:
            return None
        col, row = parsed
        value = ws[f"{col}{row}"].value
        if isinstance(value, str):
            text = value.replace("\xa0", " ").strip()
            try:
                return float(text.replace(',', '.'))
            except Exception:
                return text
        return value

    def _normalize_lookup_key(self, value: Any) -> Optional[str]:
        if value is None:
            return None
        if isinstance(value, (int, float)):
            return f"{float(value):.10g}"
        text = str(value).replace("\xa0", " ").strip()
        text = re.sub(r"\s+", " ", text)
        try:
            num = float(text.replace(',', '.'))
            return f"{num:.10g}"
        except Exception:
            return text.lower()

    def _sap_cache_has_values(self, cache: dict) -> bool:
        col_d = cache.get('col_D', {})
        col_c = cache.get('col_C_to_M', {})
        has_d = any(v not in (None, "") for v in col_d.values())
        has_c = any(cols and cols[0] not in (None, "") for cols in col_c.values())
        return has_d and has_c

    def _get_range_columns(self, range_text: str) -> Optional[Tuple[str, str, Optional[int], Optional[int]]]:
        clean = range_text.replace('$', '')
        if ':' not in clean:
            return None
        start, end = clean.split(':', 1)
        start_match = re.match(r"([A-Za-z]+)(\d+)", start)
        end_match = re.match(r"([A-Za-z]+)(\d+)", end)
        if not start_match or not end_match:
            return None
        start_col, start_row = start_match.group(1).upper(), start_match.group(2)
        end_col, end_row = end_match.group(1).upper(), end_match.group(2)
        return start_col, end_col, int(start_row) if start_row else None, int(end_row) if end_row else None

    def _resolve_sheet_from_ref(self, ref: str, sheet_ref_name: str, ws_source, ws_dest):
        if '!' not in ref:
            return ws_source, ref
        sheet_part, range_part = ref.split('!', 1)
        sheet_name = sheet_part.strip("'")
        if sheet_name == sheet_ref_name:
            return ws_source, range_part
        if sheet_name.lower() == "mzlz consolidated":
            return ws_dest, range_part
        return ws_source, range_part

    def _get_column_values(self, ws, col: str, start_row: Optional[int] = None, end_row: Optional[int] = None) -> List:
        max_row = ws.max_row
        start = start_row or 1
        end = end_row or max_row
        values = []
        for row in range(start, end + 1):
            values.append(ws[f"{col}{row}"].value)
        return values

    def _safe_eval(self, expr: str) -> Optional[float]:
        try:
            node = ast.parse(expr, mode='eval')
        except Exception:
            return None

        operators = {
            ast.Add: operator.add,
            ast.Sub: operator.sub,
            ast.Mult: operator.mul,
            ast.Div: operator.truediv,
            ast.USub: operator.neg,
            ast.UAdd: operator.pos,
        }

        def _eval(node):
            if isinstance(node, ast.Expression):
                return _eval(node.body)
            if isinstance(node, ast.Num):
                return node.n
            if isinstance(node, ast.Constant) and isinstance(node.value, (int, float)):
                return node.value
            if isinstance(node, ast.BinOp) and type(node.op) in operators:
                return operators[type(node.op)](_eval(node.left), _eval(node.right))
            if isinstance(node, ast.UnaryOp) and type(node.op) in operators:
                return operators[type(node.op)](_eval(node.operand))
            raise ValueError("Unsupported expression")

        try:
            return _eval(node)
        except Exception:
            return None

    def _evaluate_formula(self, formula: str, ws_source, ws_dest, sheet_ref_name: str, source_column: str) -> Optional[
        float]:
        if not formula:
            return None
        text = formula.strip()
        if text.startswith('='):
            text = text[1:]
        if text.startswith('+'):
            text = text[1:]

        upper = text.upper()
        if upper.startswith("SUMIFS("):
            args_text = text[len("SUMIFS("):-1]
            args = self._split_formula_args(args_text)
            if len(args) < 3:
                return None
            sum_ws, sum_range = self._resolve_sheet_from_ref(args[0], sheet_ref_name, ws_source, ws_dest)
            crit_ws, crit_range = self._resolve_sheet_from_ref(args[1], sheet_ref_name, ws_source, ws_dest)
            crit_val = self._get_cell_value_from_ws(ws_dest, args[2].replace('$', ''))
            if crit_val is None:
                try:
                    crit_val = float(args[2].replace(',', '.').strip('"'))
                except Exception:
                    crit_val = args[2].strip('"')
            range_info = self._get_range_columns(sum_range)
            crit_info = self._get_range_columns(crit_range)
            if not range_info or not crit_info:
                return None
            sum_col, _, sum_start, sum_end = range_info
            crit_col, _, crit_start, crit_end = crit_info
            sum_values = self._get_column_values(sum_ws, sum_col, sum_start, sum_end)
            crit_values = self._get_column_values(crit_ws, crit_col, crit_start, crit_end)
            total = 0
            for sum_val, crit_val_cell in zip(sum_values, crit_values):
                if crit_val_cell == crit_val:
                    try:
                        total += float(sum_val or 0)
                    except Exception:
                        pass
            return total

        if upper.startswith("SUMIF("):
            args_text = text[len("SUMIF("):-1]
            args = self._split_formula_args(args_text)
            if len(args) < 2:
                return None
            crit_ws, crit_range = self._resolve_sheet_from_ref(args[0], sheet_ref_name, ws_source, ws_dest)
            crit_val = self._get_cell_value_from_ws(ws_dest, args[1].replace('$', ''))
            if crit_val is None:
                try:
                    crit_val = float(args[1].replace(',', '.').strip('"'))
                except Exception:
                    crit_val = args[1].strip('"')
            sum_range_text = args[2] if len(args) >= 3 else args[0]
            sum_ws, sum_range = self._resolve_sheet_from_ref(sum_range_text, sheet_ref_name, ws_source, ws_dest)
            crit_info = self._get_range_columns(crit_range)
            sum_info = self._get_range_columns(sum_range)
            if not crit_info or not sum_info:
                return None
            crit_col, _, crit_start, crit_end = crit_info
            sum_col, _, sum_start, sum_end = sum_info
            crit_values = self._get_column_values(crit_ws, crit_col, crit_start, crit_end)
            sum_values = self._get_column_values(sum_ws, sum_col, sum_start, sum_end)
            total = 0
            for sum_val, crit_val_cell in zip(sum_values, crit_values):
                if crit_val_cell == crit_val:
                    try:
                        total += float(sum_val or 0)
                    except Exception:
                        pass
            return total

        if upper.startswith("VLOOKUP("):
            args_text = text[len("VLOOKUP("):-1]
            args = self._split_formula_args(args_text)
            if len(args) < 3:
                return None
            lookup_val = self._get_cell_value_from_ws(ws_dest, args[0].replace('$', ''))
            if lookup_val is None:
                try:
                    lookup_val = float(args[0].replace(',', '.').strip('"'))
                except Exception:
                    lookup_val = args[0].strip('"')
            table_ws, table_range = self._resolve_sheet_from_ref(args[1], sheet_ref_name, ws_source, ws_dest)
            col_index = int(float(args[2]))
            range_info = self._get_range_columns(table_range)
            if not range_info:
                return None
            start_col, end_col, start_row, end_row = range_info
            start_idx = column_index_from_string(start_col)
            end_idx = column_index_from_string(end_col)
            max_row = end_row or table_ws.max_row
            for row in range(start_row or 1, max_row + 1):
                key = table_ws[f"{start_col}{row}"].value
                if key == lookup_val:
                    target_idx = start_idx + col_index - 1
                    if target_idx > end_idx:
                        return None
                    target_col = get_column_letter(target_idx)
                    return table_ws[f"{target_col}{row}"].value
            return None

        # Fallback: arithmetic expression with destination sheet references
        expr = text

        def replace_cell(match):
            cell_ref = match.group(0)
            value = self._get_cell_value_from_ws(ws_dest, cell_ref.replace('$', ''))
            return str(value or 0)

        expr = re.sub(r"\$[A-Za-z]+\$\d+", replace_cell, expr)
        return self._safe_eval(expr)

    def _build_sap_cache(self, ws_source) -> dict:
        """Pre-read SAP sheet columns A, C-M and D into dicts for fast SUMIFS/VLOOKUP.
        Single pass through the sheet    called once per transfer card.
        """
        cache = {'col_A': {}, 'col_D': {}, 'col_C_to_M': {}}
        if ws_source is None:
            return cache
        row_num = 1
        try:
            row_iter = ws_source.iter_rows(min_row=1, max_row=ws_source.max_row, values_only=True)
            use_values_only = True
        except TypeError:
            row_iter = ws_source.iter_rows()
            use_values_only = False

        for row in row_iter:
            if not use_values_only:
                row = [cell.value for cell in row]
            col_a = row[0] if len(row) > 0 else None
            col_d = row[3] if len(row) > 3 else None
            cache['col_A'][row_num] = col_a
            cache['col_D'][row_num] = col_d
            if len(row) >= 13:
                cache['col_C_to_M'][row_num] = list(row[2:13])  # columns C..M
            row_num += 1
        print(f"[DEBUG] SAP cache built: {row_num - 1} rows")
        return cache

    def _sumifs_from_cache(self, cache: dict, criteria_val) -> float:
        """Fast SUMIFS using pre-built cache: sum col D where col A == criteria_val."""
        total = 0.0
        col_a = cache.get('col_A', {})
        col_d = cache.get('col_D', {})
        norm_crit = self._normalize_lookup_key(criteria_val)
        for row_num, a_val in col_a.items():
            if a_val is None:
                continue
            if self._normalize_lookup_key(a_val) == norm_crit:
                try:
                    total += float(col_d.get(row_num) or 0)
                except Exception:
                    pass
        return total

    def _vlookup_from_cache(self, cache: dict, lookup_val, col_index: int = 2) -> object:
        """Fast VLOOKUP using pre-built cache: find lookup_val in col C, return col at col_index offset."""
        norm_lookup = self._normalize_lookup_key(lookup_val)
        if norm_lookup is None:
            return None
        for row_num, cols in cache.get('col_C_to_M', {}).items():
            if cols and self._normalize_lookup_key(cols[0]) == norm_lookup:
                idx = col_index - 1  # col_index=2 -> index 1 (col D)
                if 0 <= idx < len(cols):
                    return cols[idx]
        return None

    def _apply_fixed_formula_values(
            self,
            ws_dest,
            dest_column: str,
            sheet_ref_name: str,
            source_column: str,
            ws_source,
            sap_cache: dict = None,
            allow_ytd_override: bool = True
    ) -> int:
        if not dest_column:
            return 0

        # The fixed formulas always reference the month's SAP sheet.
        # SAP sheets are named by month number: "1", "2", ... "12"
        # or the new format "MM_YYYY" e.g. "08_2026".
        # If sheet_ref_name is non-numeric (e.g. "REFI 2025", "BUDGET 2025"),
        # we use ws_source to find the correct sheet name from the loaded SAP workbook.
        import re as _re
        _sap_sheet_pattern = _re.compile(r'^\d{1,2}(:_\d{4})$')
        if not sheet_ref_name or not _sap_sheet_pattern.match(str(sheet_ref_name)):
            # Try to get the actual sheet name from ws_source's parent workbook
            if ws_source is not None:
                try:
                    actual_sheet_name = ws_source.title
                    if _sap_sheet_pattern.match(str(actual_sheet_name)):
                        sheet_ref_name = actual_sheet_name
                        print(
                            f"[DEBUG] _apply_fixed_formula_values: resolved sheet_ref_name from ws_source.title = '{sheet_ref_name}'")
                    else:
                        # ws_source is not a SAP sheet (e.g. it's a Cost Control sheet)
                        # This happens for REFI/BUDGET    sheet_ref_name stays as-is
                        # and the formula evaluation will use ws_source directly
                        print(
                            f"[DEBUG] _apply_fixed_formula_values: ws_source.title='{actual_sheet_name}' is not SAP sheet, keeping sheet_ref_name='{sheet_ref_name}'")
                except Exception:
                    pass

        month_sheet = self._format_sheet_name_for_formula(str(sheet_ref_name))

        # Build SAP cache once    avoids re-reading entire sheet for every formula row
        if sap_cache is None:
            sap_cache = self._build_sap_cache(ws_source)

        # Helper: get dest cell value
        def dest_val(row):
            return self._get_cell_value_from_ws(ws_dest, f"C{row}")

        # SUMIFS rows: sum col D where col A == $C{row}
        sumifs_rows = {57, 118, 119, 120, 121, 122, 123}
        # VLOOKUP rows: lookup $C{row} in col C:M, return col 2 (col D), divide by -1000
        vlookup_rows_neg = {115, 116, 117, 135, 142, 143, 144, 145, 146, 147}

        if not allow_ytd_override:
            # Do not override rows 118-123 and 142-147 for non-SAP cards
            sumifs_rows = {r for r in sumifs_rows if r not in {118,119,120,121,122,123}}
            vlookup_rows_neg = {r for r in vlookup_rows_neg if r not in {142,143,144,145,146,147}}
        # VLOOKUP rows: divide by +1000
        vlookup_rows_pos = {210, 212, 216}

        repaired_count = 0

        for row in sorted(sumifs_rows | vlookup_rows_neg | vlookup_rows_pos | {58, 65}):
            cell = ws_dest[f"{dest_column}{row}"]

            if row in sumifs_rows:
                crit = dest_val(row)
                val = self._sumifs_from_cache(sap_cache, crit) / -1000.0
                cell.value = val
                repaired_count += 1

            elif row == 58:
                # SUMIFS on col C for 3 criteria
                c57 = dest_val(57)
                c62 = dest_val(62)
                c67 = dest_val(67)
                col_c = sap_cache.get('col_C_to_M', {})
                col_d = sap_cache.get('col_D', {})
                total = 0.0
                for rn, cols in col_c.items():
                    if cols:
                        c_val = str(cols[0]).strip() if cols[0] is not None else None
                        if c_val in (str(c57).strip(), str(c62).strip(), str(c67).strip()):
                            try:
                                total += float(col_d.get(rn) or 0)
                            except Exception:
                                pass
                cell.value = total / -1000.0
                repaired_count += 1

            elif row == 65:
                # SUMIF on col C for 2 criteria
                c65 = dest_val(65)
                c66 = dest_val(66)
                col_c = sap_cache.get('col_C_to_M', {})
                col_d = sap_cache.get('col_D', {})
                total = 0.0
                for rn, cols in col_c.items():
                    if cols:
                        c_val = str(cols[0]).strip() if cols[0] is not None else None
                        if c_val in (str(c65).strip(), str(c66).strip()):
                            try:
                                total += float(col_d.get(rn) or 0)
                            except Exception:
                                pass
                cell.value = total / -1000.0
                repaired_count += 1

            elif row in vlookup_rows_neg:
                lv = dest_val(row)
                raw = self._vlookup_from_cache(sap_cache, lv, col_index=2)
                try:
                    cell.value = float(raw) / -1000.0 if raw is not None else 0
                except Exception:
                    cell.value = 0
                repaired_count += 1

            elif row in vlookup_rows_pos:
                lv = dest_val(row)
                raw = self._vlookup_from_cache(sap_cache, lv, col_index=2)
                try:
                    cell.value = float(raw) / 1000.0 if raw is not None else 0
                except Exception:
                    cell.value = 0
                repaired_count += 1

        if repaired_count:
            print(f"[DEBUG] Fixed formula values written in column {dest_column}: {repaired_count}")
        return repaired_count

    def _copy_defined_names_for_sheet(self, src_wb, dest_wb, src_sheet_name: str, dest_sheet_name: str) -> None:
        try:
            dest_sheet = dest_wb[dest_sheet_name]
            dest_idx = dest_wb.sheetnames.index(dest_sheet_name)
        except Exception:
            return

        try:
            from openpyxl.workbook.defined_name import DefinedName
        except Exception:
            DefinedName = None

        for defined in list(getattr(src_wb.defined_names, "definedName", [])):
            try:
                destinations = list(defined.destinations)
            except Exception:
                destinations = []
            if not destinations:
                continue
            if not any(sheet == src_sheet_name for sheet, _ in destinations):
                continue

            attr_text = defined.attr_text
            if attr_text:
                attr_text = self._replace_sheet_reference(attr_text, src_sheet_name, dest_sheet_name)

            try:
                if dest_wb.defined_names.get(defined.name, dest_sheet if defined.localSheetId is not None else None):
                    continue
            except Exception:
                pass

            if DefinedName:
                new_def = DefinedName(defined.name, attr_text=attr_text, comment=defined.comment)
            else:
                new_def = copy(defined)
                new_def.attr_text = attr_text

            new_def.hidden = getattr(defined, "hidden", None)
            if defined.localSheetId is not None:
                new_def.localSheetId = dest_idx
            dest_wb.defined_names.append(new_def)

    def _copy_full_sheet(self, mappings, deferred_save=False):
        """Full sheet copy - copies entire sheet from source to destination"""
        print(f"[DEBUG] ========== _copy_full_sheet START ==========")
        print(f"[DEBUG] _copy_full_sheet called for {mappings.get('month')} {mappings.get('year')}")
        print(f"[DEBUG] mappings keys: {list(mappings.keys())}")

        month_year = (mappings['month'], mappings['year'])
        read_type = mappings.get('source_file_type', 'cost_control')
        write_month = mappings.get('write_month', month_year[0])
        write_year = mappings.get('write_year', month_year[1])

        read_key = (month_year[0], month_year[1], read_type)
        write_key = (write_month, write_year, 'cost_control')

        if read_key not in self.loaded_workbooks:
            print(f"[WARN] Read workbook not loaded: {read_key} (skipping transfer)")
            return 0

        if write_key not in self.loaded_workbooks:
            print(f"[WARN] Write workbook not loaded: {write_key} (skipping transfer)")
            return 0

        read_wb = self.loaded_workbooks[read_key]['workbook']
        write_wb = self.loaded_workbooks[write_key]['workbook']
        write_path = self.loaded_workbooks[write_key]['path']

        src_sheet_name = mappings['source_sheet'].strip()
        target_name = mappings.get('custom_sheet_name') or src_sheet_name or "CopiedSheet"

        print(f"[DEBUG] _copy_full_sheet: src={src_sheet_name}, target={target_name}")

        # Check if target sheet already exists - skip if it does
        if target_name in write_wb.sheetnames:
            print(f"[_copy_full_sheet] Skipped - sheet {target_name} already exists in destination")
            return 0

        # Resolve source sheet
        resolved_sheet = src_sheet_name

        def normalize_sheet(name):
            return re.sub(r'[^a-z0-9]', '', name.lower())

        if resolved_sheet not in read_wb.sheetnames:
            candidates = {normalize_sheet(s.strip()): s for s in read_wb.sheetnames}
            trimmed = src_sheet_name.replace('Worksheet ', '').strip()
            normalized = normalize_sheet(trimmed)

            if normalized in candidates:
                resolved_sheet = candidates[normalized]
            elif len(read_wb.sheetnames) == 1:
                resolved_sheet = read_wb.sheetnames[0]

        if resolved_sheet not in read_wb.sheetnames:
            print(f"[_copy_full_sheet] ERROR: Sheet '{src_sheet_name}'   in source")
            return 0

        # Create new sheet in destination
        src_ws = read_wb[resolved_sheet]
        dest_ws = write_wb.create_sheet(target_name)

        sheet_ref_name = mappings.get('source_sheet') or src_ws.title
        source_column = mappings.get('source_column', 'A')

        # Copy all cells from source to destination
        from copy import copy
        from openpyxl.styles import PatternFill, Font, Alignment, Border, Side

        for row in src_ws.iter_rows():
            for cell in row:
                cell_coordinate = getattr(cell, 'coordinate', None)
                if not cell_coordinate:
                    continue
                dest_cell = dest_ws[cell_coordinate]

                # Copy value (rewrite sheet references and repair #REF formulas)
                if cell.data_type == 'f' and isinstance(cell.value, str):
                    updated_formula = self._replace_sheet_reference(cell.value, src_ws.title, dest_ws.title)
                    updated_formula = self._repair_ref_formula(updated_formula, sheet_ref_name, source_column)
                    dest_cell.value = updated_formula
                else:
                    dest_cell.value = cell.value

                # Copy basic styling
                if cell.has_style:
                    dest_cell.font = copy(cell.font)
                    dest_cell.border = copy(cell.border)
                    dest_cell.fill = copy(cell.fill)
                    dest_cell.number_format = copy(cell.number_format)
                    dest_cell.protection = copy(cell.protection)
                    dest_cell.alignment = copy(cell.alignment)

        # Copy column dimensions (if available)
        try:
            for col in getattr(src_ws, "column_dimensions", {}):
                if col in src_ws.column_dimensions:
                    dest_ws.column_dimensions[col].width = src_ws.column_dimensions[col].width
        except Exception:
            pass

        # Copy row dimensions (if available)
        try:
            for row in getattr(src_ws, "row_dimensions", {}):
                if row in src_ws.row_dimensions:
                    dest_ws.row_dimensions[row].height = src_ws.row_dimensions[row].height
        except Exception:
            pass

        # Copy defined names that point to the source sheet (avoid #REF in formulas)
        self._copy_defined_names_for_sheet(read_wb, write_wb, src_ws.title, dest_ws.title)

        print(f" Full sheet copied: {src_sheet_name}{target_name}")

        # Save the workbook after copying sheet
        write_wb.save(write_path)
        print(f"[DEBUG] Workbook saved after sheet copy")

        if not deferred_save:
            break_external_links(write_path, wb_values=self.loaded_workbooks.get(write_key, {}).get('wb_values'))
            self._finalize_with_excel_calc(write_path)

        return src_ws.max_row * src_ws.max_column

    def closeEvent(self, event):
        """Clean shutdown - prevents ZipFile errors"""
        # Close all loaded workbooks
        for wb_info in list(self.loaded_workbooks.values()):
            try:
                wb_info['workbook'].close()
            except:
                pass
        self.loaded_workbooks.clear()

        # Also close any workbooks that might be open in the individual tab
        if hasattr(self, 'indiv_panel'):
            try:
                if hasattr(self.indiv_panel, 'source_wb') and self.indiv_panel.source_wb:
                    self.indiv_panel.source_wb.close()
                if hasattr(self.indiv_panel, 'dest_wb') and self.indiv_panel.dest_wb:
                    self.indiv_panel.dest_wb.close()
            except:
                pass

        event.accept()

    def _load_selected_months(self):
        """Load files for all selected months and years."""
        selected_months = [item.text() for item in self.month_list.selectedItems()]
        selected_years = [int(item.text()) for item in self.year_list.selectedItems()]

        if not selected_months:
            QMessageBox.warning(self, "No Selection", "Please select at least one month.")
            return

        if not selected_years:
            QMessageBox.warning(self, "No Selection", "Please select at least one year.")
            return

        self._clear_all_mapping_rows()

        for wb_info in self.loaded_workbooks.values():
            try:
                wb_info['workbook'].close()
            except:
                pass
        self.loaded_workbooks = {}

        files_loaded = []
        files_missing = []
        files_errors = []

        # Build list of (month, year, file_path, filename) to load
        load_tasks = []
        for year in sorted(selected_years):
            for month in selected_months:
                month_num = MONTH_TO_NUM[month]
                base = f"Cost Control ZAG {month_num}_{year}_working"
                file_path, filename = None, base + ".xlsm"
                for ext in (".xlsm", ".xlsx"):
                    candidate = self.source_folder / (base + ext)
                    if candidate.is_file():
                        file_path, filename = candidate, base + ext
                        break
                if file_path is not None:
                    load_tasks.append((month, year, file_path, filename))
                else:
                    files_missing.append((month, year, base + ".xlsm"))

        def _load_one(task):
            """Load a single workbook (and its data_only twin) in a thread."""
            month, year, file_path, filename = task
            try:
                wb = openpyxl.load_workbook(str(file_path), keep_vba=True)
                wb_values = openpyxl.load_workbook(str(file_path), data_only=True, keep_vba=True)
                return ('ok', month, year, filename, wb, wb_values, str(file_path))
            except Exception as e:
                return ('error', month, year, filename, str(e))

        # Load all files in parallel (I/O bound   threads genuinely help)
        import concurrent.futures
        max_workers = min(len(load_tasks), 4)  # cap at 4 to avoid hammering network drives
        with concurrent.futures.ThreadPoolExecutor(max_workers=max(max_workers, 1)) as pool:
            futures = {pool.submit(_load_one, task): task for task in load_tasks}
            for future in concurrent.futures.as_completed(futures):
                result = future.result()
                if result[0] == 'ok':
                    _, month, year, filename, wb, wb_values, path = result
                    key = (month, year, 'cost_control')
                    self.loaded_workbooks[key] = {
                        'workbook': wb,
                        'wb_values': wb_values,  #  cached data_only twin, no re-load in _transfer
                        'path': path,
                        'filename': filename,
                    }
                    # Also store under legacy key for backwards compatibility
                    self.loaded_workbooks[(month, year)] = self.loaded_workbooks[key]
                    files_loaded.append((month, year, filename))
                    print(f"[LOAD] {filename}")
                else:
                    _, month, year, filename, err = result
                    files_errors.append((month, year, f"{filename} (Error: {err})"))
                    print(f"[LOAD]  {filename}: {err}")

        self._update_files_table(files_loaded, files_missing, files_errors)

        if not files_loaded:
            self.month_status.setText("No files found")
            self.month_status.setStyleSheet("color:red; font-weight:bold;")

                                f"No files found for selected months/years.\n\n"
                                f"Looking in folder:\n{self.source_folder}\n\n"
                                "Check the 'Show Details' section for missing files.")

        self.month_status.setText(f"Loaded {len(files_loaded)} file(s)")
        self.month_status.setStyleSheet("color:#065F46; font-weight:bold;")

        self.is_modified = False
        self.btn_save.setEnabled(False)

                self._toggle_files_list()

        self._display_mappings_for_selected_months(selected_months, selected_years)
        Toast(self, f"Loaded {len(files_loaded)} file(s)")

        # Show the load summary dialog
        self._show_load_summary_dialog(files_loaded, files_missing, files_errors)

    def _apply_quarter_to_all(self, quarter: int, replace: bool = False):
        """
        Apply (check) the months of the given quarter to ALL existing PeriodRow widgets.

        Args:
            quarter: 1-4

                     If False   add the quarter without removing other checked months
        """
        if self.periods_layout.count() == 0:
            self.period_status.setText("No period rows yet    generate first")
            self.period_status.setStyleSheet("color: #D97706;")
            return

        indices = QUARTERS.get(quarter, [])
        if not indices:
            self.period_status.setText(f"Invalid quarter {quarter}")
            self.period_status.setStyleSheet("color: #DC2626;")
            return

        affected_rows = 0

        # Make a copy to avoid potential issues if layout changes during iteration
        items = [self.periods_layout.itemAt(i) for i in range(self.periods_layout.count())]

        for item in items:
            if item is None:
                continue
            widget = item.widget()
            if not isinstance(widget, PeriodRow):
                continue

            if replace:
                widget._clear()

            month_list = list(widget.month_checkboxes.values())
            for idx in indices:
                if 0 <= idx < len(month_list):
                    month_list[idx].setChecked(True)
            affected_rows += 1

        action = "set" if replace else "added"
        self.period_status.setText(f"Q{quarter} {action} to {affected_rows} year(s)")
        self.period_status.setStyleSheet("color: #059669; font-style: italic;")

    def _add_mapping_with_password(self):
        if not self._verify_password("add a mapping"):
            return
        # Use first selected month/year from list widgets
        selected_months = [item.text() for item in self.month_list.selectedItems()]
        selected_years = [int(item.text()) for item in self.year_list.selectedItems()]
        month_name = selected_months[0] if selected_months else None
        year = selected_years[0] if selected_years else None
        self._add_mapping_row(month_name=month_name, year=year)

    def _remove_mapping_with_password(self, row):
        if not self._verify_password("remove this mapping"):
            return
        self._remove_mapping_row(row)

    def _remove_mapping_row(self, row):
        if row in self.mapping_rows:
            self._remove_mapping_from_group(row)
            row.deleteLater()
            self.mapping_rows.remove(row)
            self._renumber_mappings()
            self._save_mappings_to_json()

    def _delete_selected_mappings(self):
        selected = [r for r in self.mapping_rows if r.is_selected()]
        if not selected:
            QMessageBox.information(self, "Info", "No mappings selected.")
            return
        if not self._verify_password(f"delete {len(selected)} mapping(s)"):
            return
        for row in selected:
            self._remove_mapping_from_group(row)
            row.deleteLater()
            self.mapping_rows.remove(row)
        self._renumber_mappings()
        self._save_mappings_to_json()
        Toast(self, f"Deleted {len(selected)} mapping(s)")

    def _renumber_mappings(self):
        for i, row in enumerate(self.mapping_rows, start=1):
            row.update_index(i)
        self.mapping_counter = len(self.mapping_rows)

    def _save_mappings_to_json(self) -> None:
        for row in self.mapping_rows:
            mapping_dict = row.get_mapping_dict()
            month = mapping_dict.get('month', '').lower()
            target_file = self.mappings_files_by_type.get(
                getattr(row, "source_file_type", "cost_control"),
                self.current_mappings_file
            )
            all_mappings = self._load_mappings_from_file(target_file)
            found = False
            for i, entry in enumerate(all_mappings):
                if entry.get('month', '').lower() == month:
                    all_mappings[i] = mapping_dict
                    found = True
                    break
            if not found and month:
                all_mappings.append(mapping_dict)

            try:
                with open(target_file, "w", encoding="utf-8") as f:
                    json.dump(all_mappings, f, indent=2, ensure_ascii=False)
            except Exception as e:
                print(f"Failed to save {Path(target_file).name}: {e}")

    def _add_staff_mapping_rows(self, month_name: str, year: int) -> int:
        """
        Add mapping row for staff FTE data.
        Source: Staff_{year}.xlsx -> sheet "{year} FTE" -> row 33 -> column D-O (based on month)
        Dest:   MZLZ Consolidated -> row 33 -> column G/W/AM/BD/BW/CP/DJ/EF/FB/FY/GX/HW (based on month)
        """
        # Column mapping: month -> source column in {year} FTE sheet (D-O)
        MONTH_TO_STAFF_SOURCE_COL = {
            'January': 'D', 'February': 'E', 'March': 'F',
            'April': 'G', 'May': 'H', 'June': 'I',
            'July': 'J', 'August': 'K', 'September': 'L',
            'October': 'M', 'November': 'N', 'December': 'O',
        }
        
        source_col = MONTH_TO_STAFF_SOURCE_COL.get(month_name)
        if not source_col:
            return 0
        
        # Read from JSON file to get destination column
        staff_file = MAPPINGS_STAFF_FILE
        staff_mappings = self._load_mappings_from_file(staff_file)
        staff_matched = next(
            (e for e in staff_mappings if str(e.get("month", "")).lower().strip() == month_name.lower().strip()),
            None
        )
        
        if not staff_matched:
            print(f"[DEBUG] No staff mapping found for {month_name}")
            return 0
        
        # Get destination info from JSON
        dest_sheet = str(staff_matched.get("dest_sheet", "MZLZ Consolidated"))
        dest_column = str(staff_matched.get("dest_column"))
        dest_row = int(staff_matched.get("dest_row", 33))
        
        if not dest_column:
            print(f"[DEBUG] No dest_column in staff mapping for {month_name}")
            return 0
        
        # Get source sheet from JSON and replace {year} placeholder
        source_sheet_template = str(staff_matched.get("source_sheet", "{year} FTE"))
        source_sheet = source_sheet_template.replace("{year}", str(year))
        
        # Access the Staff workbook for this month/year
        staff_key = (month_name, year, "staff")
        if staff_key not in self.loaded_workbooks:
            print(f"[DEBUG] Staff workbook not loaded for {month_name} {year}")
            return 0
        
        staff_wb_info = self.loaded_workbooks[staff_key]
        staff_wb = staff_wb_info['workbook']
        
        # Check if sheet exists
        if source_sheet not in staff_wb.sheetnames:
            print(f"[DEBUG] Sheet '{source_sheet}' not found in Staff workbook. Available: {staff_wb.sheetnames}")
            return 0
        
        # Create the mapping row
        self._add_mapping_row(
            source_sheet=source_sheet,
            source_column=source_col,
            source_rows=[33],  # Always row 33
            dest_sheet=dest_sheet,
            dest_column=dest_column,
            dest_rows=[dest_row],  # Always row 33
            month_name=month_name,
            year=year,
            mappings_file=staff_file,
            source_file_type="staff",
            save_json=False
        )
        
        return 1

    #  EXCEL FINALIZE (xlwings recalc + save)

    def _finalize_with_excel_calc(self, file_path: str) -> None:
        """
        Open the saved file in a hidden Excel instance, force a full recalculation,
        then save and close. This ensures:
          1. All formula cells get fresh <v> cached values reflecting our new data.
          2. Cross-sheet formula chains (IF, SUM, VLOOKUP etc.) propagate correctly.
          3. The file opens instantly in Excel with no blank cells and no F9 needed.

        Non-fatal: if xlwings/Excel is unavailable (e.g. no Excel installed),
        we log a warning and continue    the break_external_links injection still
        provides a best-effort result.
        """
        try:
            print(f"[finalize] Opening in Excel for recalc: {Path(file_path).name}")
            app = xw.App(visible=False, add_book=False)
            app.display_alerts = False
            app.screen_updating = False
            try:
                app.api.EnableEvents = False
            except Exception:
                pass
            try:
                wb = app.books.open(file_path)
                # Force full recalculation of all sheets
                app.calculate()
                wb.save()
                try:
                    app.api.EnableEvents = True
                except Exception:
                    pass
                wb.close()
                print(f"[finalize] Excel recalc + save done: {Path(file_path).name}")
            finally:
                try:
                    app.quit()
                except Exception:
                    pass
        except Exception as e:
            print(f"[finalize] xlwings recalc skipped (non-fatal): {e}")

    def _resolve_dest_row_by_label(
            self,
            ws_dest,
            expected_row: int,
            expected_label: str,
            missing_labels: List[str],
            search_radius: int = 20
    ) -> Optional[int]:
        def normalize_label(value: str) -> str:
            return " ".join(str(value).strip().lower().split())

        expected_norm = normalize_label(expected_label)
        if not expected_norm:
            return expected_row

        current_label = ws_dest[f"B{expected_row}"].value
        if current_label is not None and normalize_label(current_label) == expected_norm:
            return expected_row

        for offset in range(-search_radius, search_radius + 1):
            candidate = expected_row + offset
            if candidate <= 0:
                continue
            value = ws_dest[f"B{candidate}"].value
            if value is not None and normalize_label(value) == expected_norm:
                return candidate

        missing_labels.append(f"Row {expected_row} -> '{expected_label}'")
        return None

    def execute_single(self, row):
        if not self.loaded_workbooks:
            QMessageBox.warning(self, "Not Loaded", "Load file(s) first!")
            return
        self.missing_destination_labels = []
        try:
            m = row.get_values()
            if not m['source_sheet'] or not m['source_column'] or not m['dest_sheet'] or not m['dest_column']:
                raise ValueError("Sheet and column fields are required.")
            if not m['source_rows'] or not m['dest_rows']:
                raise ValueError("No row mappings defined. Click 'Edit Rows' to add.")

            cells = self._transfer(m)
            if cells > 0:
                self.fly_airplane()
                Toast(self, f"{m.get('month', '')} {m.get('year', '')}: {cells} cells (÷ -1000)")
                QMessageBox.information(
                    self,
                    "Run Complete",
                    f"Transfer finished for {m.get('month', '')} {m.get('year', '')}.\nCells transferred: {cells}."
                )
            else:
                QMessageBox.information(self, "Info", "No data in source cells.")

            if self.missing_destination_labels:
                QMessageBox.warning(
                    self,
                    "Missing destination labels",
                    "Some destination labels were   within   20 rows:\n\n"
                    + "\n".join(sorted(set(self.missing_destination_labels)))
                )
        except Exception as e:
            QMessageBox.critical(self, "Error", str(e))

    def execute_all(self):
        if not self.loaded_workbooks:
            QMessageBox.warning(self, "Not Loaded", "Load file(s) first!")
            return
        if not self.mapping_rows:
            QMessageBox.warning(self, "No Mappings", "No mappings to execute.")
            return

        # ONLY run the cards the user has checked
        selected_rows = [row for row in self.mapping_rows if row.is_selected()]

        if not selected_rows:
            QMessageBox.warning(
                self,
                "Nothing Selected",
                "Please tick the checkbox on the left of each card you want to transfer.\n\n"
                "Unchecked cards will be ignored."
            )
            return

        # Prevent execution if any selected files are open in Excel
        def _is_locked(path):
            if not path or not os.path.exists(path):
                return False
            try:
                with open(path, "a"):
                    return False
            except (PermissionError, OSError):
                return True

        locked_paths = []
        for row in selected_rows:
            m = row.get_values()
            read_key = (m.get('month'), m.get('year'), m.get('source_file_type', 'cost_control'))
            write_key = (m.get('write_month', m.get('month')), m.get('write_year', m.get('year')), 'cost_control')
            for key in (read_key, write_key):
                entry = self.loaded_workbooks.get(key)
                if entry:
                    path = entry.get('path')
                    if _is_locked(path) and path not in locked_paths:
                        locked_paths.append(path)

        if locked_paths:
            QMessageBox.warning(
                self,
                "Close Excel files",
                "Please close the following Excel files before running:\n\n" + "\n".join(locked_paths)
            )
            return

        self.missing_destination_labels = []
        self._transfer_paused = False

        #   Launch TransferWorker in a background thread
        self._transfer_worker = TransferWorker(self, selected_rows)
        self._transfer_thread = QThread()
        self._transfer_worker.moveToThread(self._transfer_thread)

        self._transfer_thread.started.connect(self._transfer_worker.run)
        self._transfer_worker.progress.connect(self._on_transfer_progress)
        self._transfer_worker.row_done.connect(self._on_transfer_row_done)
        self._transfer_worker.finished.connect(self._on_transfer_finished)
        self._transfer_worker.finished.connect(self._transfer_thread.quit)
        self._transfer_worker.finished.connect(self._transfer_worker.deleteLater)
        self._transfer_thread.finished.connect(self._transfer_thread.deleteLater)

        # Disable Run, enable Stop/Pause
        self.btn_execute_all.setEnabled(False)
        self.btn_stop_transfer.setEnabled(True)
        self.btn_pause_transfer.setEnabled(True)
        self.period_status.setText("Running transfers")

        self._transfer_thread.start()

            return
        window_width = self.width()
        self.airplane_label.move(-100, 150)
        self.airplane_label.show()
        self.airplane_label.raise_()

        anim1 = QPropertyAnimation(self.airplane_label, b"pos")
        anim1.setDuration(800)
        anim1.setStartValue(QPoint(-100, 150))
        anim1.setEndValue(QPoint(window_width // 3, 80))
        anim1.setEasingCurve(QEasingCurve.Type.OutQuad)

        anim2 = QPropertyAnimation(self.airplane_label, b"pos")
        anim2.setDuration(600)
        anim2.setStartValue(QPoint(window_width // 3, 80))
        anim2.setEndValue(QPoint(2 * window_width // 3, 80))
        anim2.setEasingCurve(QEasingCurve.Type.Linear)

        anim3.setStartValue(QPoint(2 * window_width // 3, 80))
        anim3.setEndValue(QPoint(window_width + 100, 30))
        anim3.setEasingCurve(QEasingCurve.Type.InQuad)

        self.animation_group.addAnimation(anim1)
        self.animation_group.addAnimation(anim2)
        self.animation_group.addAnimation(anim3)
        self.animation_group.finished.connect(self.airplane_label.hide)
        self.animation_group.start()

                self, "Unsaved Changes", "Save before closing",
                QMessageBox.StandardButton.Yes | QMessageBox.StandardButton.No | QMessageBox.StandardButton.Cancel
            )
            if reply == QMessageBox.StandardButton.Yes:
                self.save_file()
            elif reply == QMessageBox.StandardButton.Cancel:
                event.ignore()
                return

        for wb_info in self.loaded_workbooks.values():
            try:
                wb_info['workbook'].close()
            except:
                pass
        event.accept()

    def _load_mappings_from_file(self, filepath: Path) -> List[Dict]:
        """Load mappings from a specific JSON file (safe version)"""
        if not filepath.is_file():
            print(f"Warning: {filepath.name}  , using empty mappings")
            return []
        try:
            with open(filepath, "r", encoding="utf-8-sig") as f:
                data = json.load(f)
                return data if isinstance(data, list) else []
        except Exception as e:
            print(f"Error Loading {filepath.name}: {e}")
            return []

    def _load_sheets(self):
        try:
            wb = openpyxl.load_workbook(self.file_path, read_only=True)
            self.sheet_combo.clear()
            self.sheet_combo.addItems(wb.sheetnames)
            wb.close()
            if self.sheet_combo.count() > 0:
                self.sheet_combo.setCurrentIndex(0)
                self._load_current_sheet()
        except Exception as e:
            QMessageBox.critical(self, "Error", f"Could not read file:\n{str(e)}")

    def _load_current_sheet(self):
        if self.sheet_combo.count() == 0:
            return
        sheet_name = self.sheet_combo.currentText()
        self.selected_sheet_name = sheet_name
        self._create_large_scrollable_grid(sheet_name)

    def _create_large_scrollable_grid(self, sheet_name):
        """Create 5000-row grid with visible empty rows + overlay real data"""
        MAX_ROWS = 5000
        MAX_COLS = 60

        self.table.setColumnCount(MAX_COLS)
        self.table.setRowCount(MAX_ROWS)

        # Fill with empty items so grid is visible and scrollable
        for r in range(MAX_ROWS):
            for c in range(MAX_COLS):
                self.table.setItem(r, c, QTableWidgetItem(""))

        # Column headers
        for c in range(MAX_COLS):
            self.table.setHorizontalHeaderItem(c, QTableWidgetItem(get_column_letter(c + 1)))

        # Overlay real data from the sheet
        try:
            wb = openpyxl.load_workbook(self.file_path, read_only=True, data_only=True)
            ws = wb[sheet_name]
            data = list(ws.iter_rows(min_row=1, max_row=MAX_ROWS, max_col=MAX_COLS, values_only=True))

            for r, row in enumerate(data):
                for c, value in enumerate(row):
                    if value is not None:
                        self.table.setItem(r, c, QTableWidgetItem(str(value)))
            wb.close()
        except:
            pass

        self.table.itemSelectionChanged.connect(self._on_selection_changed)
        self.table.resizeColumnsToContents()

    def _on_selection_changed(self):
        self.selected_cells = []
        for index in self.table.selectedIndexes():
            row = index.row() + 1
            col = index.column() + 1
            self.selected_cells.append((row, col))

        self.selected_cells = sorted(set(self.selected_cells))
        self.status_label.setText(f"Selected cells: {len(self.selected_cells)}")

    def accept(self):
        if not self.selected_cells:
            QMessageBox.warning(self, "No Selection", "Please select at least one cell.")
            return
        super().accept()

    def _add_source_to_mapping(self, cells, sheet_name):
        for r, c in cells:
            cell_ref = get_column_letter(c) + str(r)
            self.mapping_data.append([cell_ref, 0.0, "", sheet_name])  # temp value
        self._refresh_mapping_table()

    def _add_dest_to_mapping(self, cells, sheet_name):
        # Match existing rows or append
        for i, cell in enumerate(cells):
            if i < len(self.mapping_data):
                r, c = cell
                self.mapping_data[i][2] = get_column_letter(c) + str(r)
            else:
                self.mapping_data.append(["", 0.0, get_column_letter(c) + str(r), sheet_name])
        self._refresh_mapping_table()

    def _refresh_mapping_table(self):
        self.mapping_table.setRowCount(len(self.mapping_data))
        for i, row_data in enumerate(self.mapping_data):
            self.mapping_table.setItem(i, 0, QTableWidgetItem(str(row_data[0])))
            self.mapping_table.setItem(i, 1, QTableWidgetItem(str(row_data[1])))
            arrow = QTableWidgetItem("")
            arrow.setTextAlignment(Qt.AlignmentFlag.AlignCenter)
            arrow.setFlags(Qt.ItemFlag.ItemIsEnabled)
            self.mapping_table.setItem(i, 2, arrow)
            self.mapping_table.setItem(i, 3, QTableWidgetItem(str(row_data[2])))

        self.mapping_counter.setText(f"Mapped cells: {len(self.mapping_data)}")
        self._fix_arrows_and_sync()

    def _on_rows_about_to_move(self, parent, start, end, destination, row):
        self._drag_in_progress = True

    def _on_rows_moved(self, parent, start, end, destination, row):
        """Called automatically after user drags a row"""
        QTimer.singleShot(10, self._fix_arrows_and_sync)  # small delay = Qt finished moving

    def _on_item_edited(self, item):
        if self._drag_in_progress:
            return
        if item and item.column() != 2:
            self._sync_mapping_data_from_table()

    def _fix_arrows_and_sync(self):
        """Re-insert clean arrows + sync data + clear any leftover selection"""
        self.mapping_table.blockSignals(True)

        # Force arrow column to always show clean " "
        for r in range(self.mapping_table.rowCount()):
            arrow = QTableWidgetItem("")
            arrow.setTextAlignment(Qt.AlignmentFlag.AlignCenter)
            arrow.setFlags(Qt.ItemFlag.ItemIsEnabled)  # not selectable, not editable
            arrow.setBackground(QColor("#F8FAFC"))  # light gray background (nice look)
            self.mapping_table.setItem(r, 2, arrow)

        self.mapping_table.blockSignals(False)

        self._sync_mapping_data_from_table()
        # Remove blue selection leftovers (this was causing your first screenshot)
        self.mapping_table.clearSelection()
        self.mapping_table.setCurrentIndex(self.mapping_table.model().index(-1, -1))  # deselect all

    def _sync_mapping_data_from_table(self):
        new_data = []
        for r in range(self.mapping_table.rowCount()):
            src = self.mapping_table.item(r, 0).text().strip() if self.mapping_table.item(r, 0) else ""
            val_item = self.mapping_table.item(r, 1)
            dest = self.mapping_table.item(r, 3).text().strip() if self.mapping_table.item(r, 3) else ""

            try:
                value = float(val_item.text().strip()) if val_item and val_item.text().strip() else 0.0
            except (ValueError, TypeError):
                value = 0.0

            sheet = ""
            for col in (0, 1, 3):
                item = self.mapping_table.item(r, col)
                if item and item.data(Qt.ItemDataRole.UserRole):
                    sheet = item.data(Qt.ItemDataRole.UserRole)
                    break
            new_data.append([src, value, dest, sheet])

        self.mapping_data = new_data
        self.mapping_counter.setText(f"Mapped cells: {len(self.mapping_data)}")

    def _add_mapping_row(self):
        self.mapping_data.append(["", 0.0, "", ""])
        self._refresh_mapping_table()

    def _clear_mapping_table(self):
        if QMessageBox.question(self, "Clear All", "Clear entire mapping") == QMessageBox.StandardButton.Yes:
            self.mapping_data.clear()
            self._refresh_mapping_table()

    def _remove_selected_mapping(self):
        selected = self.mapping_table.selectedIndexes()
        if not selected:
            return
        rows = sorted(set(idx.row() for idx in selected), reverse=True)
        for row in rows:
            if row < len(self.mapping_data):
                del self.mapping_data[row]
        self._refresh_mapping_table()

    def _resolve_excel_path(self, path: Path) -> Path:
        if path.is_file():
            return path
        if not path.suffix:
            for ext in (".xlsx", ".xlsm", ".xls"):
                candidate = path.with_suffix(ext)
                if candidate.is_file():
                    return candidate
        return path

    def _execute(self):
        if not self.copy_sheet_checkbox.isChecked() and not self.mapping_data:
            QMessageBox.warning(self, "No Mapping", "No mappings defined.")
            return

        try:
            original_path = self._resolve_excel_path(Path(self.dest_path))
            if not original_path.is_file():
                raise FileNotFoundError(f"Destination file  :\n{original_path}")
            if original_path.name.startswith(""):
                raise ValueError("Please select the real Excel file, not the temporary file.")

            # Write directly using Excel (xlwings) to preserve file integrity
            app = xw.App(visible=False, add_book=False)
            app.display_alerts = False
            app.screen_updating = False
            try:
                app.api.EnableEvents = False
            except Exception:
                pass
            try:
                if any(os.path.samefile(book.fullname, str(original_path)) for book in app.books):
                    raise RuntimeError("Destination file is already open in Excel. Please close it and try again.")

                wb = app.books.open(str(original_path))
                dest_sheet = self.dest_sheet_name if hasattr(self, 'dest_sheet_name') and self.dest_sheet_name else None
                ws_dest = wb.sheets[dest_sheet] if dest_sheet else wb.sheets.active

                transferred = 0
                if self.copy_sheet_checkbox.isChecked():
                    src_path = self._resolve_excel_path(Path(self.src_path))
                    if not src_path.is_file():
                        raise FileNotFoundError(f"Source file  :\n{src_path}")

                    wb_src = app.books.open(str(src_path))
                    try:
                        src_sheet = self.source_sheet_name if hasattr(self,
                                                                      'source_sheet_name') and self.source_sheet_name else None
                        ws_src = wb_src.sheets[src_sheet] if src_sheet else wb_src.sheets.active
                        used = ws_src.api.UsedRange
                        max_row = used.Rows.Count
                        max_col = used.Columns.Count
                        ws_dest.range((1, 1), (max_row, max_col)).value = ws_src.range((1, 1), (max_row, max_col)).value
                        transferred = max_row * max_col
                    finally:
                        wb_src.close()
                else:
                    for row_idx in range(self.mapping_table.rowCount()):
                        src_item = self.mapping_table.item(row_idx, 0)
                        value_item = self.mapping_table.item(row_idx, 1)
                        dest_item = self.mapping_table.item(row_idx, 3)
                        if not src_item or not dest_item:
                            continue
                        dest_ref = dest_item.text().strip()
                        if not dest_ref:
                            continue

                        value_text = value_item.text().strip() if value_item else ""
                        value = None
                        if value_text != "":
                            try:
                                value = float(value_text)
                            except ValueError:
                                value = value_text

                        ws_dest.range(dest_ref).value = value
                        transferred += 1

                wb.save()
                wb.close()

            finally:
                app.quit()

            QMessageBox.information(self, "Success",
                                    f"Transfer completed successfully!\n\n"
                                    f"{transferred} cells updated\n\n"
                                    f"File saved:\n"
                                    f"{original_path.name}")

            self.main.fly_airplane()

        except Exception as e:
            QMessageBox.critical(self, "Error", f"Transfer failed:\n{str(e)}\n\n"
                                                f"Original file is safe (backup was created).")

                    json.dump(all_mappings, f, indent=2, ensure_ascii=False)
            except Exception as e:
                print(f"Failed to save {Path(target_file).name}: {e}")

    #  EXCEL FINALIZE (xlwings recalc + save)

    def _finalize_with_excel_calc(self, file_path: str) -> None:
        """
        Open the saved file in a hidden Excel instance, force a full recalculation,
        then save and close. This ensures:
          1. All formula cells get fresh <v> cached values reflecting our new data.
          2. Cross-sheet formula chains (IF, SUM, VLOOKUP etc.) propagate correctly.
          3. The file opens instantly in Excel with no blank cells and no F9 needed.

        Non-fatal: if xlwings/Excel is unavailable (e.g. no Excel installed),
        we log a warning and continue    the break_external_links injection still
        provides a best-effort result.
        """
        try:
            print(f"[finalize] Opening in Excel for recalc: {Path(file_path).name}")
            app = xw.App(visible=False, add_book=False)
            app.display_alerts = False
            app.screen_updating = False
            try:
                app.api.EnableEvents = False
            except Exception:
                pass
            try:
                wb = app.books.open(file_path)
                # Force full recalculation of all sheets
                app.calculate()
                wb.save()
                try:
                    app.api.EnableEvents = True
                except Exception:
                    pass
                wb.close()
                print(f"[finalize] Excel recalc + save done: {Path(file_path).name}")
            finally:
                try:
                    app.quit()
                except Exception:
                    pass
        except Exception as e:
            print(f"[finalize] xlwings recalc skipped (non-fatal): {e}")

        def normalize_sheet(name):
            return re.sub(r"[^a-z0-9]", "", name.lower())

        if resolved_sheet not in read_wb.sheetnames:
            candidates = {normalize_sheet(s.strip()): s for s in read_wb.sheetnames}
            trimmed = src_sheet_name.replace("Worksheet ", "").strip()
            normalized = normalize_sheet(trimmed)

            if normalized in candidates:
                resolved_sheet = candidates[normalized]
            else:
                # Try alternate budget name formats: "2026 Budget"   "BUDGET 2026" etc.
                budget_match = re.match(r'^(\d{4})\s*budget$', trimmed, re.IGNORECASE)
                if budget_match:
                    yr = budget_match.group(1)
                    for alt in [f"{yr} Budget", f"Budget {yr}", f"BUDGET {yr}",
                                f"{yr}Budget", f"budget {yr}", f"{yr} budget"]:
                        if normalize_sheet(alt) in candidates:
                            resolved_sheet = candidates[normalize_sheet(alt)]
                            break

                # If still  , try partial match (sheet contains the key parts)
                if resolved_sheet not in read_wb.sheetnames:
                    norm_src = normalize_sheet(trimmed)
                    for sheet in read_wb.sheetnames:
                        if norm_src in normalize_sheet(sheet) or normalize_sheet(sheet) in norm_src:
                            resolved_sheet = sheet
                            print(f"[_copy_full_sheet] Partial match: '{src_sheet_name}''{sheet}'")
                            break

            if resolved_sheet not in read_wb.sheetnames and len(read_wb.sheetnames) == 1:
                resolved_sheet = read_wb.sheetnames[0]

        if resolved_sheet not in read_wb.sheetnames:
            available = ', '.join(f"'{s}'" for s in read_wb.sheetnames)
            raise ValueError(
                f"Sheet '{src_sheet_name}'   in source workbook.\n"
                f"Available sheets: {available}"
            )

        src_ws = read_wb[resolved_sheet]
        dest_ws = write_wb.create_sheet(target_name)

        # Use pre-cached data_only workbook    avoids reLoading from disk on every copy
        src_wb_values = self.loaded_workbooks[read_key].get('wb_values')
        if src_wb_values is None:
            source_path = self.loaded_workbooks[read_key]['path']
            try:
                src_wb_values = openpyxl.load_workbook(source_path, data_only=True, keep_vba=True)
                self.loaded_workbooks[read_key]['wb_values'] = src_wb_values
                print(f"[DEBUG] wb_values cache miss (copy_full_sheet)  loaded fresh")
            except Exception as e:
                print(f"[DEBUG] Could not load data_only workbook: {e}")
        try:
            src_ws_values = src_wb_values[resolved_sheet] if src_wb_values else None
        except Exception:
            src_ws_values = None

        from copy import copy
        from openpyxl.styles import (PatternFill, Font, Alignment, Border, Side,
                                     numbers as xl_numbers)

        #  Pre-cache data_only rows for value lookup
        values_rows_cache = {}
        if src_ws_values:
            for ri, vrow in enumerate(src_ws_values.iter_rows()):
                values_rows_cache[ri] = vrow

        # Copy cells: values + structure only, NO fill colors
        NO_FILL = PatternFill(fill_type=None)  # completely transparent / no fill

        for row_idx, row in enumerate(src_ws.iter_rows()):
            for col_idx, cell in enumerate(row):
                try:
                    has_style = hasattr(cell, 'has_style') and cell.has_style
                except Exception:
                    has_style = False

                # Resolve value: prefer cached (avoids copying formulas)
                cell_value = cell.value
                if cell.data_type == 'f':
                    cached = None
                    if row_idx in values_rows_cache:
                        try:
                            cached = values_rows_cache[row_idx][col_idx].value
                        except Exception:
                            cached = None
                    cell_value = cached  # None clears the formula    that's fine

                if cell_value is not None or has_style:
                    new_cell = dest_ws[cell.coordinate]
                    new_cell.value = cell_value
                    if has_style:
                        # Copy font (bold/italic/size/color) and structure    strip fill
                        new_cell.font = copy(cell.font)
                        new_cell.border = copy(cell.border)
                        new_cell.fill = NO_FILL  #  strip all colors
                        new_cell.number_format = cell.number_format
                        new_cell.alignment = copy(cell.alignment)

        # Copy merged cells and column widths
        for merged in src_ws.merged_cells.ranges:
            dest_ws.merge_cells(str(merged))
        for col in src_ws.column_dimensions:
            dest_ws.column_dimensions[col].width = src_ws.column_dimensions[col].width

        # Close the temporary values workbook
        if src_wb_values:
            src_wb_values.close()

        #   Apply Report sheet design
        # Only applied when the source sheet name contains "Report"
        if 'report' in src_sheet_name.lower():
            _apply_report_design(dest_ws)

        if not deferred_save:
            write_wb.calculation.calcMode = 'manual'
            print(f"[DEBUG] Saving workbook, type: {type(write_wb).__name__}")
            write_wb.save(write_path)
            write_wb.calculation.calcMode = 'auto'

        print(f"Full sheet copied: {src_sheet_name}{target_name}")
        return len(list(src_ws.iter_rows())) * 10

    def _resolve_dest_row_by_label(
            self,
            ws_dest,
            expected_row: int,
            expected_label: str,
            missing_labels: List[str],
            search_radius: int = 20
    ) -> Optional[int]:
        def normalize_label(value: str) -> str:
            return " ".join(str(value).strip().lower().split())

        expected_norm = normalize_label(expected_label)
        if not expected_norm:
            return expected_row

        current_label = ws_dest[f"B{expected_row}"].value
        if current_label is not None and normalize_label(current_label) == expected_norm:
            return expected_row

        for offset in range(-search_radius, search_radius + 1):
            candidate = expected_row + offset
            if candidate <= 0:
                continue
            value = ws_dest[f"B{candidate}"].value
            if value is not None and normalize_label(value) == expected_norm:
                return candidate

        missing_labels.append(f"Row {expected_row} -> '{expected_label}'")
        return None

    def execute_single(self, row):
        if not self.loaded_workbooks:
            QMessageBox.warning(self, "Not Loaded", "Load file(s) first!")
            return
        self.missing_destination_labels = []
        try:
            m = row.get_values()
            if not m['source_sheet'] or not m['source_column'] or not m['dest_sheet'] or not m['dest_column']:
                raise ValueError("Sheet and column fields are required.")
            if not m['source_rows'] or not m['dest_rows']:
                raise ValueError("No row mappings defined. Click 'Edit Rows' to add.")

            cells = self._transfer(m)
            if cells > 0:
                self.fly_airplane()
                Toast(self, f"{m.get('month', '')} {m.get('year', '')}: {cells} cells (-1000)")
                QMessageBox.information(
                    self,
                    "Run Complete",
                    f"Transfer finished for {m.get('month', '')} {m.get('year', '')}.\nCells transferred: {cells}."
                )
            else:
                QMessageBox.information(self, "Info", "No data in source cells.")

            if self.missing_destination_labels:
                QMessageBox.warning(
                    self,
                    "Missing destination labels",
                    "Some destination labels were   within   20 rows:\n\n"
                    + "\n".join(sorted(set(self.missing_destination_labels)))
                )
        except Exception as e:
            QMessageBox.critical(self, "Error", str(e))

    def execute_all(self):
        if not self.loaded_workbooks:
            QMessageBox.warning(self, "Not Loaded", "Load file(s) first!")
            return
        if not self.mapping_rows:
            QMessageBox.warning(self, "No Mappings", "No mappings to execute.")
            return

        # ONLY run the cards the user has checked
        selected_rows = [row for row in self.mapping_rows if row.is_selected()]

        if not selected_rows:
            QMessageBox.warning(
                self,
                "Nothing Selected",
                "Please tick the checkbox on the left of each card you want to transfer.\n\n"
                "Unchecked cards will be ignored."
            )
            return

        # Prevent execution if any selected files are open in Excel
        def _is_locked(path):
            if not path or not os.path.exists(path):
                return False
            try:
                with open(path, "a"):
                    return False
            except (PermissionError, OSError):
                return True

        locked_paths = []
        for row in selected_rows:
            m = row.get_values()
            read_key = (m.get('month'), m.get('year'), m.get('source_file_type', 'cost_control'))
            write_key = (m.get('write_month', m.get('month')), m.get('write_year', m.get('year')), 'cost_control')
            for key in (read_key, write_key):
                entry = self.loaded_workbooks.get(key)
                if entry:
                    path = entry.get('path')
                    if _is_locked(path) and path not in locked_paths:
                        locked_paths.append(path)

        if locked_paths:
            QMessageBox.warning(
                self,
                "Close Excel files",
                "Please close the following Excel files before running:\n\n" + "\n".join(locked_paths)
            )
            return

        self.missing_destination_labels = []
        self._transfer_paused = False

        #   Launch TransferWorker in a background thread
        self._transfer_worker = TransferWorker(self, selected_rows)
        self._transfer_thread = QThread()
        self._transfer_worker.moveToThread(self._transfer_thread)

        self._transfer_thread.started.connect(self._transfer_worker.run)
        self._transfer_worker.progress.connect(self._on_transfer_progress)
        self._transfer_worker.row_done.connect(self._on_transfer_row_done)
        self._transfer_worker.finished.connect(self._on_transfer_finished)
        self._transfer_worker.finished.connect(self._transfer_thread.quit)
        self._transfer_worker.finished.connect(self._transfer_worker.deleteLater)
        self._transfer_thread.finished.connect(self._transfer_thread.deleteLater)

        # Disable Run, enable Stop/Pause
        self.btn_execute_all.setEnabled(False)
        self.btn_stop_transfer.setEnabled(True)
        self.btn_pause_transfer.setEnabled(True)
        self.period_status.setText("Running transfers")

        self._transfer_thread.start()

    def closeEvent(self, event):

        if self.is_modified:
            reply = QMessageBox.question(
                self, "Unsaved Changes", "Save before closing",
                QMessageBox.StandardButton.Yes | QMessageBox.StandardButton.No | QMessageBox.StandardButton.Cancel
            )
            if reply == QMessageBox.StandardButton.Yes:

                wb_info['workbook'].close()

                pass
        event.accept()

    def _load_sheets(self):
        try:
            wb = openpyxl.load_workbook(self.file_path, read_only=True)
            self.sheet_combo.clear()
            self.sheet_combo.addItems(wb.sheetnames)
            wb.close()
            if self.sheet_combo.count() > 0:
                self.sheet_combo.setCurrentIndex(0)
                self._load_current_sheet()
        except Exception as e:
            QMessageBox.critical(self, "Error", f"Could not read file:\n{str(e)}")

    def _load_current_sheet(self):
        if self.sheet_combo.count() == 0:
            return
        sheet_name = self.sheet_combo.currentText()
        self.selected_sheet_name = sheet_name
        self._create_large_scrollable_grid(sheet_name)

    def _create_large_scrollable_grid(self, sheet_name):
        """Create 5000-row grid with visible empty rows + overlay real data"""
        MAX_ROWS = 5000
        MAX_COLS = 60

        self.table.setColumnCount(MAX_COLS)
        self.table.setRowCount(MAX_ROWS)

        # Fill with empty items so grid is visible and scrollable
        for r in range(MAX_ROWS):
            for c in range(MAX_COLS):
                self.table.setItem(r, c, QTableWidgetItem(""))

        # Column headers
        for c in range(MAX_COLS):
            self.table.setHorizontalHeaderItem(c, QTableWidgetItem(get_column_letter(c + 1)))

        # Overlay real data from the sheet
        try:
            wb = openpyxl.load_workbook(self.file_path, read_only=True, data_only=True)
            ws = wb[sheet_name]
            data = list(ws.iter_rows(min_row=1, max_row=MAX_ROWS, max_col=MAX_COLS, values_only=True))

            for r, row in enumerate(data):
                for c, value in enumerate(row):
                    if value is not None:
                        self.table.setItem(r, c, QTableWidgetItem(str(value)))
            wb.close()
        except:
            pass

        self.table.itemSelectionChanged.connect(self._on_selection_changed)
        self.table.resizeColumnsToContents()

    def _on_selection_changed(self):
        self.selected_cells = []
        for index in self.table.selectedIndexes():
            row = index.row() + 1
            col = index.column() + 1
            self.selected_cells.append((row, col))

        self.selected_cells = sorted(set(self.selected_cells))
        self.status_label.setText(f"Selected cells: {len(self.selected_cells)}")

    def accept(self):
        if not self.selected_cells:
            QMessageBox.warning(self, "No Selection", "Please select at least one cell.")
            return
        super().accept()

    def _browse_source(self):
        path, _ = QFileDialog.getOpenFileName(self, "Source Excel File", "", "Excel (*.xlsx *.xlsm *.xls)")
        if path:
            self.source_path = path
            self.src_path_label.setText(path)

    def _browse_dest(self):
        path, _ = QFileDialog.getOpenFileName(self, "Destination Excel File", "", "Excel (*.xlsx *.xlsm *.xls)")
        if path:
            self.dest_path = path
            self.dest_path_label.setText(path)

    def _load_and_select_source(self):
        if not self.source_path:
            QMessageBox.warning(self, "No File", "Please select source file first.")
            return
        dialog = SheetCellSelectorDialog(self, self.source_path, "Source")
        if dialog.exec() == QDialog.DialogCode.Accepted:
            self._add_source_to_mapping(dialog.selected_cells, dialog.selected_sheet_name)

    def _load_and_select_dest(self):
        if not self.dest_path:
            QMessageBox.warning(self, "No File", "Please select destination file first.")
            return
        dialog = SheetCellSelectorDialog(self, self.dest_path, "Destination")
        if dialog.exec() == QDialog.DialogCode.Accepted:
            self._add_dest_to_mapping(dialog.selected_cells, dialog.selected_sheet_name)

    def _add_source_to_mapping(self, cells, sheet_name):
        for r, c in cells:
            cell_ref = get_column_letter(c) + str(r)
            self.mapping_data.append([cell_ref, 0.0, "", sheet_name])  # temp value
        self._refresh_mapping_table()

    def _add_dest_to_mapping(self, cells, sheet_name):
        # Match existing rows or append
        for i, cell in enumerate(cells):
            if i < len(self.mapping_data):
                r, c = cell
                self.mapping_data[i][2] = get_column_letter(c) + str(r)
            else:
                self.mapping_data.append(["", 0.0, get_column_letter(c) + str(r), sheet_name])
        self._refresh_mapping_table()

    def _refresh_mapping_table(self):
        self.mapping_table.setRowCount(len(self.mapping_data))
        for i, row_data in enumerate(self.mapping_data):
            self.mapping_table.setItem(i, 0, QTableWidgetItem(str(row_data[0])))
            self.mapping_table.setItem(i, 1, QTableWidgetItem(str(row_data[1])))
            arrow = QTableWidgetItem("")
            arrow.setTextAlignment(Qt.AlignmentFlag.AlignCenter)
            arrow.setFlags(Qt.ItemFlag.ItemIsEnabled)
            self.mapping_table.setItem(i, 2, arrow)
            self.mapping_table.setItem(i, 3, QTableWidgetItem(str(row_data[2])))

        self.mapping_counter.setText(f"Mapped cells: {len(self.mapping_data)}")
        self._fix_arrows_and_sync()

    def _on_rows_about_to_move(self, parent, start, end, destination, row):
        self._drag_in_progress = True

    def _on_rows_moved(self, parent, start, end, destination, row):
        """Called automatically after user drags a row"""
        QTimer.singleShot(10, self._fix_arrows_and_sync)  # small delay = Qt finished moving

    def _on_item_edited(self, item):
        if self._drag_in_progress:
            return
        if item and item.column() != 2:
            self._sync_mapping_data_from_table()

    def _fix_arrows_and_sync(self):
        """Re-insert clean arrows + sync data + clear any leftover selection"""
        self.mapping_table.blockSignals(True)

        # Force arrow column to always show clean " "
        for r in range(self.mapping_table.rowCount()):
            arrow = QTableWidgetItem("")
            arrow.setTextAlignment(Qt.AlignmentFlag.AlignCenter)
            arrow.setFlags(Qt.ItemFlag.ItemIsEnabled)  # not selectable, not editable
            arrow.setBackground(QColor("#F8FAFC"))  # light gray background (nice look)
            self.mapping_table.setItem(r, 2, arrow)

        self.mapping_table.blockSignals(False)

        self._sync_mapping_data_from_table()
        # Remove blue selection leftovers (this was causing your first screenshot)
        self.mapping_table.clearSelection()
        self.mapping_table.setCurrentIndex(self.mapping_table.model().index(-1, -1))  # deselect all

    def _sync_mapping_data_from_table(self):
        new_data = []
        for r in range(self.mapping_table.rowCount()):
            src = self.mapping_table.item(r, 0).text().strip() if self.mapping_table.item(r, 0) else ""
            val_item = self.mapping_table.item(r, 1)
            dest = self.mapping_table.item(r, 3).text().strip() if self.mapping_table.item(r, 3) else ""

            try:
                value = float(val_item.text().strip()) if val_item and val_item.text().strip() else 0.0
            except (ValueError, TypeError):
                value = 0.0

            sheet = ""
            for col in (0, 1, 3):
                item = self.mapping_table.item(r, col)
                if item and item.data(Qt.ItemDataRole.UserRole):
                    sheet = item.data(Qt.ItemDataRole.UserRole)
                    break
            new_data.append([src, value, dest, sheet])

        self.mapping_data = new_data
        self.mapping_counter.setText(f"Mapped cells: {len(self.mapping_data)}")

    def _add_mapping_row(self):
        self.mapping_data.append(["", 0.0, "", ""])
        self._refresh_mapping_table()

    def _clear_mapping_table(self):
        if QMessageBox.question(self, "Clear All", "Clear entire mapping") == QMessageBox.StandardButton.Yes:
            self.mapping_data.clear()
            self._refresh_mapping_table()

    def _remove_selected_mapping(self):
        selected = self.mapping_table.selectedIndexes()
        if not selected:
            return
        rows = sorted(set(idx.row() for idx in selected), reverse=True)
        for row in rows:
            if row < len(self.mapping_data):
                del self.mapping_data[row]
        self._refresh_mapping_table()

    def _resolve_excel_path(self, path: Path) -> Path:
        if path.is_file():
            return path
        if not path.suffix:
            for ext in (".xlsx", ".xlsm", ".xls"):
                candidate = path.with_suffix(ext)
                if candidate.is_file():
                    return candidate
        return path

    def _execute(self):
        if not self.copy_sheet_checkbox.isChecked() and not self.mapping_data:
            QMessageBox.warning(self, "No Mapping", "No mappings defined.")
            return

        try:
            original_path = self._resolve_excel_path(Path(self.dest_path))
            if not original_path.is_file():
                raise FileNotFoundError(f"Destination file  :\n{original_path}")
            if original_path.name.startswith(""):
                raise ValueError("Please select the real Excel file, not the temporary '~$' file.")

            # Write directly using Excel (xlwings) to preserve file integrity
            app = xw.App(visible=False, add_book=False)
            app.display_alerts = False
            app.screen_updating = False
            try:
                app.api.EnableEvents = False
            except Exception:
                pass
            try:
                if any(os.path.samefile(book.fullname, str(original_path)) for book in app.books):
                    raise RuntimeError("Destination file is already open in Excel. Please close it and try again.")

                wb = app.books.open(str(original_path))
                dest_sheet = self.dest_sheet_name if hasattr(self, 'dest_sheet_name') and self.dest_sheet_name else None
                ws_dest = wb.sheets[dest_sheet] if dest_sheet else wb.sheets.active

                transferred = 0
                if self.copy_sheet_checkbox.isChecked():
                    src_path = self._resolve_excel_path(Path(self.src_path))
                    if not src_path.is_file():
                        raise FileNotFoundError(f"Source file  :\n{src_path}")

                    wb_src = app.books.open(str(src_path))
                    try:
                        src_sheet = self.source_sheet_name if hasattr(self,
                                                                      'source_sheet_name') and self.source_sheet_name else None
                        ws_src = wb_src.sheets[src_sheet] if src_sheet else wb_src.sheets.active
                        used = ws_src.api.UsedRange
                        max_row = used.Rows.Count
                        max_col = used.Columns.Count
                        ws_dest.range((1, 1), (max_row, max_col)).value = ws_src.range((1, 1), (max_row, max_col)).value
                        transferred = max_row * max_col
                    finally:
                        wb_src.close()
                else:
                    for row_idx in range(self.mapping_table.rowCount()):
                        src_item = self.mapping_table.item(row_idx, 0)
                        value_item = self.mapping_table.item(row_idx, 1)
                        dest_item = self.mapping_table.item(row_idx, 3)
                        if not src_item or not dest_item:
                            continue
                        dest_ref = dest_item.text().strip()
                        if not dest_ref:
                            continue

                        value_text = value_item.text().strip() if value_item else ""
                        value = None
                        if value_text != "":
                            try:
                                value = float(value_text)
                            except ValueError:
                                value = value_text

                        ws_dest.range(dest_ref).value = value
                        transferred += 1

                wb.save()
                wb.close()

            finally:
                app.quit()

            QMessageBox.information(self, "Success",
                                    f"Transfer completed successfully!\n\n"
                                    f"{transferred} cells updated\n\n"
                                    f"File saved:\n"
                                    f"{original_path.name}")

            self.main.fly_airplane()

        except Exception as e:
            QMessageBox.critical(self, "Error", f"Transfer failed:\n{str(e)}\n\n"
                                                f"Original file is safe (backup was created).")


class MappingPopup(QDialog):
    """Auto-opening modal popup for selecting which mappings to run"""

    def __init__(self, parent, mapping_rows):
        super().__init__(parent)
        self.setWindowTitle("Select Mappings to Transfer")
        self.resize(1280, 850)
        self.setModal(True)  # forces user to finish selection
        self.parent = parent
        self.mapping_rows = mapping_rows

        self._build_ui()

    def _build_ui(self):
        layout = QVBoxLayout(self)
        layout.setContentsMargins(24, 24, 24, 24)
        layout.setSpacing(18)

        self.setObjectName("mappingPopup")
        self.setStyleSheet(STYLE)

        # Header
        title = QLabel("Select Which Transfer Cards to Run")
        title.setStyleSheet("font-size: 20px; font-weight: 700; color: #1E40AF;")
        layout.addWidget(title)

        info = QLabel(
            "Check the cards you want to include.\n"
            "Unchecked cards will be skipped when you click Execute All Transfers."
        )
        info.setStyleSheet("background:#F0F9FF; color:#0369A1; padding:14px; border-radius:8px;")
        layout.addWidget(info)

        # Scroll area
        scroll = QScrollArea()
        scroll.setWidgetResizable(True)
        scroll.setFrameShape(QFrame.Shape.NoFrame)

        container = QWidget()
        self.popup_layout = QVBoxLayout(container)
        self.popup_layout.setAlignment(Qt.AlignmentFlag.AlignTop)
        self.popup_layout.setSpacing(12)

        self.popup_to_main_map = {}
        self.popup_year_groups = {}

        for main_row in self.mapping_rows:
            data = main_row.get_values()
            popup_row = MappingRow(
                index=data.get('index'),
                on_remove=None,
                on_run=None,
                on_change=None,
                source_sheet=data['source_sheet'],
                source_column=data['source_column'],
                source_rows=data['source_rows'],
                dest_sheet=data['dest_sheet'],
                dest_column=data['dest_column'],
                dest_rows=data['dest_rows'],
                month_name=data['month'],
                year=data['year'],
                mappings_file=Path(data['mappings_file']) if data.get('mappings_file') else None,
                source_file_type=data.get('source_file_type', 'cost_control')
            )

            if data.get('mapping_label'):
                popup_row.mappings_label.setText(data['mapping_label'])
            elif data.get('mappings_file'):
                popup_row.mappings_label.setText(f"Using: {Path(data['mappings_file']).name}")

            # Pre-check based on main row selection
            popup_row.set_selected(data.get('selected', True))

            # Disable buttons we don't need in the popup
            for btn in popup_row.findChildren(QPushButton):
                if btn.text() in ("Remove", "Run"):
                    btn.setEnabled(False)

            # Bind popup row to main row for two-way sync
            popup_row._bind_sync_peer(main_row)
            main_row._bind_sync_peer(popup_row)

            year_label = data.get('year') if data.get('year') is not None else "Unknown"
            group = self.popup_year_groups.get(year_label)
            if group is None:
                group = MappingYearGroup(year_label, container)
                self.popup_year_groups[year_label] = group
                self.popup_layout.addWidget(group)
            group.add_row(popup_row)

            self.popup_to_main_map[popup_row] = main_row

        scroll.setWidget(container)
        layout.addWidget(scroll, stretch=1)

        # Bottom buttons
        btn_layout = QHBoxLayout()
        btn_select_all = QPushButton("Select All")
        btn_select_all.setObjectName("btnSecondary")
        btn_select_all.clicked.connect(self._select_all_mappings)
        btn_layout.addWidget(btn_select_all)

        btn_clear_all = QPushButton("Deselect All")
        btn_clear_all.setObjectName("btnSecondary")
        btn_clear_all.clicked.connect(self._clear_all_mappings)
        btn_layout.addWidget(btn_clear_all)

        btn_layout.addStretch()

        btn_cancel = QPushButton("Cancel")
        btn_cancel.setObjectName("btnSecondary")
        btn_cancel.clicked.connect(self.reject)
        btn_layout.addWidget(btn_cancel)

        btn_confirm = QPushButton("Confirm Selection & Close")
        btn_confirm.setObjectName("btnPrimary")
        btn_confirm.setFixedHeight(52)
        btn_confirm.setStyleSheet("font-size: 15px; font-weight: 600;")
        btn_confirm.clicked.connect(self._confirm_selection)
        btn_layout.addWidget(btn_confirm)

        layout.addLayout(btn_layout)



    def _confirm_selection(self):
        """Sync checkboxes back to main window and close"""
        for popup_row, main_row in self.popup_to_main_map.items():
            main_row.set_selected(popup_row.is_selected())

        Toast(self.parent, f"Selection saved  {sum(1 for r in self.mapping_rows if r.is_selected())} card(s) active")
        self.accept()

    def _select_all_mappings(self):
        for popup_row in self.popup_to_main_map.keys():
            popup_row.set_selected(True)

    def _clear_all_mappings(self):
        for popup_row in self.popup_to_main_map.keys():
            popup_row.set_selected(False)

    def closeEvent(self, event):
        # Clear sync references to avoid stale peer access
        for popup_row, main_row in self.popup_to_main_map.items():
            popup_row._bind_sync_peer(None)
            main_row._bind_sync_peer(None)
        super().closeEvent(event)

            popup_row.set_selected(False)
